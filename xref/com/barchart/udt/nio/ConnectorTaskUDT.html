8   package com.barchart.udt.nio;
9   
10  import java.io.IOException;
11  import java.net.InetSocketAddress;
12  import java.util.Queue;
13  import java.util.concurrent.ConcurrentMap;
14  
15  import org.slf4j.Logger;
16  import org.slf4j.LoggerFactory;
17  
18  class ConnectorTaskUDT implements Runnable {
19  
20  	private static final Logger log = LoggerFactory
21  			.getLogger(ConnectorTaskUDT.class);
22  
23  	final SelectionKeyUDT keyUDT;
24  	final ConcurrentMap<SelectionKeyUDT, Runnable> taskMap;
25  	final Queue<SelectionKeyUDT> readyQueue;
26  	final InetSocketAddress remoteSocketAddress;
27  
28  	ConnectorTaskUDT(SelectionKeyUDT keyUDT,
29  			ConcurrentMap<SelectionKeyUDT, Runnable> taskMap,
30  			Queue<SelectionKeyUDT> readyQueue,
31  			InetSocketAddress remoteSocketAddress) {
32  		this.keyUDT = keyUDT;
33  		this.taskMap = taskMap;
34  		this.readyQueue = readyQueue;
35  		this.remoteSocketAddress = remoteSocketAddress;
36  	}
37  
38  	@Override
39  	public void run() {
40  
41  		IOException exception = null;
42  
43  		assert keyUDT.channelUDT.getChannelKind() == KindUDT.CONNECTOR;
44  
45  		log.debug("connect() start 	: socketID={} remoteSocketAddress={}", //
46  				keyUDT.socketID, remoteSocketAddress);
47  
48  		try {
49  
50  			// NOTE:
51  			// will block here
52  			// connect socket, not channel
53  			keyUDT.socketUDT.connect(remoteSocketAddress);
54  
55  		} catch (IOException e) {
56  
57  			// do not process; will be thrown later
58  			exception = e;
59  
60  		} catch (Throwable e) {
61  
62  			// unexpected - wrap; also process here
63  			log.error("run failed", e);
64  			exception = new IOException(e.getMessage());
65  
66  		} finally {
67  
68  			ChannelSocketUDT socketChannel = (ChannelSocketUDT) keyUDT.channelUDT;
69  			socketChannel.setConnectException(exception);
70  
71  			taskMap.remove(keyUDT);
72  			readyQueue.offer(keyUDT);
73  			keyUDT.selectorUDT.wakeup();
74  
75  		}
76  
77  		log.debug("connect() finish : socketID={} remoteSocketAddress={}", //
78  				keyUDT.socketID, remoteSocketAddress);
79  
80  	}
81  
82  }
