42  import java.io.IOException;
43  import java.net.ServerSocket;
44  import java.nio.channels.ServerSocketChannel;
45  import java.nio.channels.SocketChannel;
46  import java.nio.channels.spi.SelectorProvider;
47  
48  import com.barchart.udt.SocketUDT;

50  /**
51   * you must use {@link SelectorProviderUDT#openServerSocketChannel()} to obtain
52   * instance of this class; do not use JDK
53   * {@link java.nio.channels.ServerSocketChannel#open()}; <br>
54   * 
55   * example:
56   * 
57   * [code]
58   * 
59   * SelectorProvider provider = SelectorProviderUDT.DATAGRAM;
60   * 
61   * ServerSocketChannel acceptorChannel = provider.openServerSocketChannel();
62   * 
63   * ServerSocket acceptorSocket = acceptorChannel.socket();
64   * 
65   * InetSocketAddress acceptorAddress= new InetSocketAddress("localhost", 12345);
66   * 
67   * acceptorSocket.bind(acceptorAddress);
68   * 
69   * assert acceptorSocket.isBound();
70   * 
71   * SocketChannel connectorChannel = acceptorChannel.accept();
72   * 
73   * assert connectorChannel.isConnected();
74   * 
75   * [/code]
76   */
77  public class ChannelServerSocketUDT extends ServerSocketChannel implements
78  		ChannelUDT {
79  
80  	final SocketUDT serverSocketUDT;
81  
82  	ChannelServerSocketUDT(SelectorProvider provider, SocketUDT socketUDT) {
83  		super(provider);
84  		this.serverSocketUDT = socketUDT;
85  	}
86  
87  	@Override
88  	protected void implCloseSelectableChannel() throws IOException {
89  		serverSocketUDT.close();
90  	}
91  
92  	@Override
93  	protected void implConfigureBlocking(boolean block) throws IOException {
94  		serverSocketUDT.configureBlocking(block);
95  	}
96  
97  	@Override
98  	public SocketChannel accept() throws IOException {
99  		try {
100 			begin();
101 			SocketUDT socketUDT = serverSocketUDT.accept();
102 			SelectorProvider provider = provider();
103 			return new ChannelSocketUDT(provider, socketUDT);
104 		} finally {
105 			end(true);
106 		}
107 	}
108 
109 	// guarded by 'this'
110 	private ServerSocket serverSocketAdapter;
111 
112 	@Override
113 	public ServerSocket socket() {
114 		synchronized (this) {
115 			if (serverSocketAdapter == null) {
116 				try {
117 					serverSocketAdapter = new AdapterServerSocketUDT(this,
118 							serverSocketUDT);
119 				} catch (IOException e) {
120 					return null;
121 				}
122 			}
123 			return serverSocketAdapter;
124 		}
125 	}
126 
127 	@Override
128 	public SocketUDT getSocketUDT() {
129 		return serverSocketUDT;
130 	}
131 
132 	@Override
133 	public KindUDT getChannelKind() {
134 		return KindUDT.ACCEPTOR;
135 	}
136 
137 	@Override
138 	public boolean isOpenSocketUDT() {
139 		return serverSocketUDT.isOpen();
140 	}
141 
142 	//
143 
144 	@Override
145 	public String toString() {
146 		return serverSocketUDT.toString();
147 	}
