18  /**
19   * selection provider for UDT
20   * <p>
21   * note: you must use the same system-wide provider instance for the same
22   * {@link TypeUDT} of UDT channels and UDT selectors;
23   */
24  public class SelectorProviderUDT extends SelectorProvider {
25  
26  	/**
27  	 * system-wide provider instance, for {@link TypeUDT#DATAGRAM} UDT sockets
28  	 */
29  	public static final SelectorProviderUDT DATAGRAM = //
30  	new SelectorProviderUDT(TypeUDT.DATAGRAM);
31  
32  	/**
33  	 * system-wide provider instance, for {@link TypeUDT#STREAM} UDT sockets
34  	 */
35  	public static final SelectorProviderUDT STREAM = //
36  	new SelectorProviderUDT(TypeUDT.STREAM);
37  
38  	public static SelectorProviderUDT from(final TypeUDT type) {
39  		switch (type) {
40  		case DATAGRAM:
41  			return DATAGRAM;
42  		case STREAM:
43  			return STREAM;
44  		default:
45  			throw new IllegalStateException("wrong type=" + type);
46  		}
47  	}
48  
49  	/**
50  	 * {@link TypeUDT} of UDT sockets generated by this provider
51  	 */
52  	public final TypeUDT type;
53  
54  	protected SelectorProviderUDT(final TypeUDT type) {
55  		this.type = type;
56  	}
57  
58  	protected volatile int acceptQueueSize = SocketUDT.DEFAULT_ACCEPT_QUEUE_SIZE;
59  
60  	public int getAcceptQueueSize() {
61  		return acceptQueueSize;
62  	}
63  
64  	public void setAcceptQueueSize(final int queueSize) {
65  		acceptQueueSize = queueSize;
66  	}
67  
68  	protected volatile int maxSelectorSize = SocketUDT.DEFAULT_MAX_SELECTOR_SIZE;
69  
70  	public int getMaxSelectorSize() {
71  		return maxSelectorSize;
72  	}
73  
74  	public void setMaxSelectorSize(final int selectorSize) {
75  		maxSelectorSize = selectorSize;
76  	}
77  
78  	/**
79  	 * TODO support datagram
80  	 */
81  	@Override
82  	public DatagramChannel openDatagramChannel() throws IOException {
83  		throw new RuntimeException("TODO");
84  	}
85  
86  	@Override
87  	public Pipe openPipe() throws IOException {
88  		throw new RuntimeException("feature not available");
89  	}
90  
91  	@Override
92  	public SelectorUDT openSelector() throws IOException {
93  		return new SelectorUDT(this, maxSelectorSize);
94  	}
95  
96  	@Override
97  	public ChannelServerSocketUDT openServerSocketChannel() throws IOException {
98  		final SocketUDT serverSocketUDT = new SocketUDT(type);
99  		return new ChannelServerSocketUDT(this, serverSocketUDT);
100 	}
101 
102 	@Override
103 	public ChannelSocketUDT openSocketChannel() throws IOException {
104 		final SocketUDT socketUDT = new SocketUDT(type);
105 		return new ChannelSocketUDT(this, socketUDT);
106 	}
