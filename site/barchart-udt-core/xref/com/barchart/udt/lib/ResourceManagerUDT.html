23   * class path resource extractor and system loader

25  public class ResourceManagerUDT {

27  	protected final static Logger log = LoggerFactory

30  	protected static boolean isSameResource(final URLConnection conONE,
31  			final URLConnection conTWO) throws Exception {
32  
33  		final long timeONE = conONE.getLastModified();
34  		final long sizeONE = conONE.getContentLength();
35  
36  		final long timeTWO = conTWO.getLastModified();
37  		final long sizeTWO = conTWO.getContentLength();
38  
39  		return sizeONE == sizeTWO && timeONE == timeTWO;
40  
41  	}

43  	protected static URLConnection fileConnection(final File file)
44  			throws Exception {
45  
46  		final URL url = file.toURI().toURL();
47  
48  		final URLConnection connection = url.openConnection();
49  
50  		return connection;
51  
52  	}

54  	protected final static int EOF = -1;
55  
56  	/** will use time stamp of jar file */
57  	protected static long timeStamp(final URLConnection connIN) {
58  		return connIN.getLastModified();
59  	}

61  	/**
62  	 * extract resource from class path into local file system
63  	 */
64  	protected static void extractResource(final String sourcePath,
65  			final String targetPath) throws Exception {
66  
67  		final URL sourceUrl = ResourceManagerUDT.class.getResource(sourcePath);
68  
69  		if (sourceUrl == null) {
70  			log.warn("classpath resource not found: {}", sourcePath);
71  			throw new IllegalArgumentException("resource not found");
72  		}
73  
74  		log.debug("sourceURL={} ", sourceUrl);
75  
76  		final URLConnection sourceConn = sourceUrl.openConnection();
77  
78  		if (sourceConn == null) {
79  			log.warn("classpath resource connection not available: {}",
80  					sourcePath);
81  			throw new IllegalArgumentException("resource not found");
82  		}
83  
84  		final File targetFile = new File(targetPath).getAbsoluteFile();
85  		log.debug("targetFile={} ", targetFile);
86  
87  		final File targetFolder = targetFile.getParentFile().getAbsoluteFile();
88  		log.debug("targetFolder={} ", targetFolder);
89  
90  		ensureTargetFolder(targetFolder);
91  
92  		final URLConnection targetConn = fileConnection(targetFile);
93  
94  		if (isSameResource(sourceConn, targetConn)) {
95  			log.debug("already extracted; sourcePath={} targetPath={}",
96  					sourcePath, targetPath);
97  			return;
98  		} else {
99  			log.debug("make new extraction destination for targetPath={}",
100 					targetPath);
101 			targetFile.delete();
102 			targetFile.createNewFile();
103 		}
104 
105 		final long sourceTime = timeStamp(sourceConn);
106 
107 		final InputStream sourceStream = new BufferedInputStream(//
108 				sourceUrl.openStream());
109 
110 		final OutputStream targetStream = new BufferedOutputStream(//
111 				new FileOutputStream(targetFile));
112 
113 		final byte[] array = new byte[64 * 1024];
114 
115 		int readCount = 0;
116 
117 		while ((readCount = sourceStream.read(array)) != EOF) {
118 			targetStream.write(array, 0, readCount);
119 		}
120 
121 		targetStream.flush();
122 
123 		sourceStream.close();
124 		targetStream.close();
125 
126 		/** synchronize target time stamp with source to avoid repeated copy */
127 		targetFile.setLastModified(sourceTime);
128 
129 		log.debug("extracted OK; sourcePath={} targetPath={}", sourcePath,
130 				targetPath);
131 
132 	}

134 	protected static void ensureTargetFolder(final File folder)
135 			throws Exception {
136 		if (folder.exists()) {
137 			if (folder.isDirectory()) {
138 				log.debug("found folder={}", folder);
139 			} else {
140 				log.error("not a directory; folder={}", folder);
141 				throw new IllegalArgumentException(
142 						"extract destination exists, but as a file and not a folder");
143 			}
144 		} else {
145 			final boolean isSuccess = folder.mkdirs();
146 			if (isSuccess) {
147 				log.debug("mkdirs : folder={}", folder);
148 			} else {
149 				log.error("mkdirs failure; folder={}", folder);
150 				throw new IllegalStateException(
151 						"failed to make extract destination folder");
152 			}
153 		}
154 	}
155 
156 	protected static void ensureTargetFolder(final String targetFolder)
157 			throws Exception {
158 
159 		final File folder = new File(targetFolder).getAbsoluteFile();
160 
161 		ensureTargetFolder(folder);
162 
163 	}
