8   package com.barchart.udt.net;
9   
10  import java.io.IOException;
11  import java.io.OutputStream;
12  import java.nio.channels.IllegalBlockingModeException;
13  
14  import com.barchart.udt.ErrorUDT;
15  import com.barchart.udt.SocketUDT;
16  
17  /**
18   * {@link OutputStream} for UDT sockets.
19   */
20  public class NetOutputStreamUDT extends OutputStream {
21  
22  	protected final SocketUDT socketUDT;
23  
24  	/**
25  	 * 
26  	 * @param socketUDT
27  	 *            The UDT socket.
28  	 */
29  	public NetOutputStreamUDT(final SocketUDT socketUDT) {
30  
31  		if (!socketUDT.isBlocking()) {
32  			throw new IllegalBlockingModeException();
33  		}
34  
35  		this.socketUDT = socketUDT;
36  
37  	}
38  
39  	@Override
40  	public void write(final int b) throws IOException {
41  
42  		// Just cast it -- this is the same thing SocketOutputStream does.
43  		final byte[] bytes = { (byte) b };
44  
45  		write(bytes);
46  
47  	}
48  
49  	@Override
50  	public void write(final byte[] bytes) throws IOException {
51  
52  		write(bytes, 0, bytes.length);
53  
54  	}
55  
56  	@Override
57  	public void write(final byte[] bytes, final int off, final int len)
58  			throws IOException {
59  
60  		int bytesRemaining = len;
61  
62  		while (bytesRemaining > 0) {
63  
64  			final int count = socketUDT.send(bytes, off + len - bytesRemaining,
65  					off + len);
66  
67  			if (count > 0) {
68  				bytesRemaining -= count;
69  				continue;
70  			}
71  
72  			if (count == 0) {
73  				throw new ExceptionSendUDT(socketUDT.id(),
74  						ErrorUDT.USER_DEFINED_MESSAGE, "UDT send time out");
75  			}
76  
77  			throw new IllegalStateException(
78  					"Socket has been chaged to non-blocking");
79  		}
80  
81  	}
82  
83  	@Override
84  	public void close() throws IOException {
85  		socketUDT.close();
86  	}
87  
88  }
