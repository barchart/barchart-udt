545 	/**
546 	 * receive into {@link java.nio.channels.DirectByteBuffer}; upto
547 	 * {@link java.nio.ByteBuffer#remaining()} bytes
548 	 * 
549 	 * @return <code>-1</code> : nothing received (non-blocking only)<br>
550 	 *         <code>=0</code> : timeout expired (blocking only)<br>
551 	 *         <code>>0</code> : normal receive, byte count<br>
552 	 * @see #receive2(int, int, ByteBuffer, int, int)
553 	 */
554 	public final int receive(final ByteBuffer buffer) throws ExceptionUDT {
555 		checkBuffer(buffer);
556 		final int position = buffer.position();
557 		final int limit = buffer.limit();
558 		final int remaining = buffer.remaining();
559 		final int sizeReceived = receive2(socketID, socketType, //
560 				buffer, position, limit);
561 		if (sizeReceived <= 0) {
562 			return sizeReceived;
563 		}
564 		if (sizeReceived <= remaining) {
565 			buffer.position(position + sizeReceived);
566 			return sizeReceived;
567 		} else { // should not happen
568 			log.error("sizeReceived > remaining");
569 			return 0;
570 		}
571 	}
