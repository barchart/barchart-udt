343 	/**
344 	 * current monitor status snapshot for all parameters
345 	 */
346 	public void appendSnapshot(StringBuilder text) {
347 
348 		text.append("\n\t");
349 		text.append("socketID");
350 		text.append(" = ");
351 		text.append(socketUDT.socketID);
352 
353 		Field fieldArray[] = MonitorUDT.class.getDeclaredFields();
354 
355 		for (Field field : fieldArray) {
356 
357 			if (!isNumeric(field)) {
358 				continue;
359 			}
360 
361 			try {
362 
363 				field.setAccessible(true);
364 
365 				String fieldName = field.getName();
366 				String fieldValue = field.get(this).toString();
367 
368 				text.append("\n\t");
369 				text.append(fieldName);
370 				text.append(" = ");
371 				text.append(fieldValue);
372 
373 			} catch (Exception e) {
374 				log.error("unexpected", e);
375 			}
376 
377 		}
378 
379 		double localSendLoss = 100.0 * (double) pktSndLoss / (double) pktSent;
380 
381 		text.append("\n\t% localSendLoss = ");
382 		text.append(localSendLoss);
383 
384 		double localReceiveLoss = 100.0 * (double) pktRcvLoss
385 				/ (double) pktRecv;
386 
387 		text.append("\n\t% localReceiveLoss = ");
388 		text.append(localReceiveLoss);
389 
390 	}
