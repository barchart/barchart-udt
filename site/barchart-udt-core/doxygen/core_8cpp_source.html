00535 {

00540

00542       throw CUDTException(5, 2, 0);
00543 
00544    if (m_bConnected)
00545       throw CUDTException(5, 2, 0);
00546

00548    // RendezevousQueue is used to temporarily store incoming handshake, non-rendezvous connections also require this function
00549    m_pRcvQueue->m_pRendezvousQueue->insert(m_SocketID, m_iIPversion, serv_addr);
00550 
00551    CPacket request;
00552    char* reqdata = new char [m_iPayloadSize];
00553    CHandShake req;
00554 
00555    CPacket response;
00556    char* resdata = new char [m_iPayloadSize];

00558 
00559    // This is my current configurations
00560    req.m_iVersion = m_iVersion;
00561    req.m_iType = m_iSockType;
00562    req.m_iMSS = m_iMSS;
00563    req.m_iFlightFlagSize = (m_iRcvBufSize < m_iFlightFlagSize)? m_iRcvBufSize : m_iFlightFlagSize;

00577    m_ullSndLastAck2Time = CTimer::getTime();

00580    request.pack(0, NULL, reqdata, m_iPayloadSize);
00581    // ID = 0, connection request
00582    request.m_iID = 0;
00583

00585    response.pack(0, NULL, resdata, m_iPayloadSize);
00586

00588    if (m_bRendezvous)
00589       timeo *= 10;

01659 void CUDT::sendCtrl(const int& pkttype, void* lparam, void* rparam, const int& size)
01660 {

01662 
01663    switch (pkttype)
01664    {
01665    case 2: //010 - Acknowledgement
01666       {
01667       int32_t ack;
01668 
01669       // If there is no loss, the ACK is the current largest sequence number plus 1;
01670       // Otherwise it is the smallest sequence number in the receiver loss list.
01671       if (0 == m_pRcvLossList->getLossLength())
01672          ack = CSeqNo::incseq(m_iRcvCurrSeqNo);
01673       else
01674          ack = m_pRcvLossList->getFirstLostSeq();
01675 
01676       if (ack == m_iRcvLastAckAck)
01677          break;
01678 
01679       // send out a lite ACK
01680       // to save time on buffer processing and bandwidth/AS measurement, a lite ACK only feeds back an ACK number
01681       if (4 == size)
01682       {
01683          ctrlpkt.pack(pkttype, NULL, &ack, size);
01684          ctrlpkt.m_iID = m_PeerID;
01685          m_pSndQueue->sendto(m_pPeerAddr, ctrlpkt);
01686 
01687          break;
01688       }
01689 
01690       uint64_t currtime;
01691       CTimer::rdtsc(currtime);
01692 
01693       // There are new received packets to acknowledge, update related information.
01694       if (CSeqNo::seqcmp(ack, m_iRcvLastAck) > 0)
01695       {
01696          int acksize = CSeqNo::seqoff(m_iRcvLastAck, ack);
01697 
01698          m_iRcvLastAck = ack;
01699 
01700          m_pRcvBuffer->ackData(acksize);
01701 
01702          // signal a waiting "recv" call if there is any data available
01703          #ifndef WIN32
01704             pthread_mutex_lock(&m_RecvDataLock);
01705             if (m_bSynRecving)
01706                pthread_cond_signal(&m_RecvDataCond);
01707             pthread_mutex_unlock(&m_RecvDataLock);
01708          #else
01709             if (m_bSynRecving)
01710                SetEvent(m_RecvDataCond);
01711          #endif
01712 
01713          // acknowledge any waiting epolls to read
01714          s_UDTUnited.m_EPoll.enable_read(m_SocketID, m_sPollID);
01715       }
01716       else if (ack == m_iRcvLastAck)
01717       {
01718          if ((currtime - m_ullLastAckTime) < ((m_iRTT + 4 * m_iRTTVar) * m_ullCPUFrequency))
01719             break;
01720       }
01721       else
01722          break;
01723 
01724       // Send out the ACK only if has not been received by the sender before
01725       if (CSeqNo::seqcmp(m_iRcvLastAck, m_iRcvLastAckAck) > 0)
01726       {
01727          int32_t data[6];
01728 
01729          m_iAckSeqNo = CAckNo::incack(m_iAckSeqNo);
01730          data[0] = m_iRcvLastAck;
01731          data[1] = m_iRTT;
01732          data[2] = m_iRTTVar;
01733          data[3] = m_pRcvBuffer->getAvailBufSize();
01734          // a minimum flow window of 2 is used, even if buffer is full, to break potential deadlock
01735          if (data[3] < 2)
01736             data[3] = 2;
01737 
01738          if (currtime - m_ullLastAckTime > m_ullSYNInt)
01739          {
01740             data[4] = m_pRcvTimeWindow->getPktRcvSpeed();
01741             data[5] = m_pRcvTimeWindow->getBandwidth();
01742             ctrlpkt.pack(pkttype, &m_iAckSeqNo, data, 24);
01743 
01744             CTimer::rdtsc(m_ullLastAckTime);
01745          }
01746          else
01747          {
01748             ctrlpkt.pack(pkttype, &m_iAckSeqNo, data, 16);
01749          }
01750 
01751          ctrlpkt.m_iID = m_PeerID;
01752          m_pSndQueue->sendto(m_pPeerAddr, ctrlpkt);
01753 
01754          m_pACKWindow->store(m_iAckSeqNo, m_iRcvLastAck);
01755 
01756          ++ m_iSentACK;
01757          ++ m_iSentACKTotal;
01758       }
01759 
01760       break;
01761       }
01762 
01763    case 6: //110 - Acknowledgement of Acknowledgement
01764       ctrlpkt.pack(pkttype, lparam);
01765       ctrlpkt.m_iID = m_PeerID;
01766       m_pSndQueue->sendto(m_pPeerAddr, ctrlpkt);
01767 
01768       break;
01769 
01770    case 3: //011 - Loss Report
01771       {
01772       if (NULL != rparam)
01773       {
01774          if (1 == size)
01775          {
01776             // only 1 loss packet
01777             ctrlpkt.pack(pkttype, NULL, (int32_t *)rparam + 1, 4);
01778          }

02368       // If loss found, insert them to the receiver loss list
02369       m_pRcvLossList->insert(CSeqNo::incseq(m_iRcvCurrSeqNo), CSeqNo::decseq(packet.m_iSeqNo));
02370 
02371       // pack loss list for NAK
02372       int32_t lossdata[2];
02373       lossdata[0] = CSeqNo::incseq(m_iRcvCurrSeqNo) | 0x80000000;
02374       lossdata[1] = CSeqNo::decseq(packet.m_iSeqNo);
02375 
02376       // Generate loss report immediately.
02377       sendCtrl(3, NULL, lossdata, (CSeqNo::incseq(m_iRcvCurrSeqNo) == CSeqNo::decseq(packet.m_iSeqNo)) ? 1 : 2);
02378 
02379       int loss = CSeqNo::seqlen(m_iRcvCurrSeqNo, packet.m_iSeqNo) - 2;
02380       m_iTraceRcvLoss += loss;
02381       m_iRcvLossTotal += loss;
02382    }
02383 
02384    // This is not a regular fixed size packet...   
02385    //an irregular sized packet usually indicates the end of a message, so send an ACK immediately   
02386    if (packet.getLength() != m_iPayloadSize)   
02387       CTimer::rdtsc(m_ullNextACKTime); 
02388 
02389    // Update the current largest sequence number that has been received.
02390    // Or it is a retransmitted packet, remove it from receiver loss list.
02391    if (CSeqNo::seqcmp(packet.m_iSeqNo, m_iRcvCurrSeqNo) > 0)
02392       m_iRcvCurrSeqNo = packet.m_iSeqNo;
02393    else
02394       m_pRcvLossList->remove(packet.m_iSeqNo);
02395 
02396    return 0;
02397 }

02403

02405       return 1004;

02435          CMD5::compute(cookiestr.str().c_str(), cookie);
02436

02438             return -1;
02439       }
02440    }
02441 
02442    int32_t id = hs.m_iID;
02443 
02444    // When a peer side connects in...
02445    if ((1 == packet.getFlag()) && (0 == packet.getType()))
02446    {
02447       if ((hs.m_iVersion != m_iVersion) || (hs.m_iType != m_iSockType))
02448       {
02449          // mismatch, reject the request
02450          hs.m_iReqType = 1002;

02461

02464          if (result != 1)

02476       }
02477    }

02510       //send a "light" ACK
02511       sendCtrl(2, NULL, NULL, 4);
02512       ++ m_iLightACKCount;
02513    }
02514 
02515    // we are not sending back repeated NAK anymore and rely on the sender's EXP for retransmission
02516    //if ((m_pRcvLossList->getLossLength() > 0) && (currtime > m_ullNextNAKTime))
02517    //{

02521    //   CTimer::rdtsc(currtime);
02522    //   m_ullNextNAKTime = currtime + m_ullNAKInt;
02523    //}
02524 
02525    if (currtime > m_ullNextEXPTime)
02526    {
02527       // Haven't receive any information from the peer, is it dead?!
02528       // timeout: at least 16 expirations and must be greater than 10 seconds

02579
