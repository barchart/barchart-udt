00134 enum UDTOpt
00135 {
00136    UDT_MSS,             // the Maximum Transfer Unit
00137    UDT_SNDSYN,          // if sending is blocking
00138    UDT_RCVSYN,          // if receiving is blocking
00139    UDT_CC,              // custom congestion control algorithm
00140    UDT_FC,              // Flight flag size (window size)
00141    UDT_SNDBUF,          // maximum buffer in sending queue
00142    UDT_RCVBUF,          // UDT receiving buffer size
00143    UDT_LINGER,          // waiting for unsent data when closing
00144    UDP_SNDBUF,          // UDP sending buffer size
00145    UDP_RCVBUF,          // UDP receiving buffer size
00146    UDT_MAXMSG,          // maximum datagram message size
00147    UDT_MSGTTL,          // time-to-live of a datagram message
00148    UDT_RENDEZVOUS,      // rendezvous connection mode
00149    UDT_SNDTIMEO,        // send() timeout
00150    UDT_RCVTIMEO,        // recv() timeout
00151    UDT_REUSEADDR,       // reuse an existing port or create a new one
00152    UDT_MAXBW            // maximum bandwidth (bytes per second) that the connection can use
00153 };
