00107 int CCache::update(CCacheItem* data)
00108 {
00109    CGuard cacheguard(m_Lock);
00110 
00111    int key = data->getKey();
00112 
00113    if (key < 0)
00114       return -1;
00115 
00116    if (key >= m_iMaxSize)
00117       key %= m_iHashSize;
00118 
00119    CCacheItem* curr = NULL;
00120 
00121    for (list<list<CCacheItem*>::iterator>::iterator i = m_vHashPtr[key].begin(); i != m_vHashPtr[key].end(); ++ i)
00122    {
00123       if (*data == ***i)
00124       {
00125          // update the existing entry with the new value
00126          ***i = *data;
00127          curr = **i;
00128 
00129          // remove the current entry
00130          m_StorageList.erase(*i);
00131          m_vHashPtr[key].erase(i);
00132 
00133          // re-insert to the front
00134          m_StorageList.push_front(curr);
00135          m_vHashPtr[key].push_front(m_StorageList.begin());
00136 
00137          return 0;
00138       }
00139    }
00140 
00141    // create new entry and insert to front
00142    curr = data->clone();
00143    m_StorageList.push_front(curr);
00144    m_vHashPtr[key].push_front(m_StorageList.begin());
00145 
00146    ++ m_iCurrSize;
00147    if (m_iCurrSize >= m_iMaxSize)
00148    {
00149       CCacheItem* last_data = m_StorageList.back();
00150       int last_key = last_data->getKey() % m_iHashSize;
00151 
00152       for (list<list<CCacheItem*>::iterator>::iterator i = m_vHashPtr[last_key].begin(); i != m_vHashPtr[last_key].end(); ++ i)
00153       {
00154          if (*last_data == ***i)
00155          {
00156             m_vHashPtr[last_key].erase(i);
00157             break;
00158          }
00159       }
00160 
00161       delete last_data;
00162       m_StorageList.pop_back();
00163       -- m_iCurrSize;
00164    }
00165 
00166    return 0;
