00350 void CSndUList::insert_(const int64_t& ts, const CUDT* u)
00351 {
00352    CSNode* n = u->m_pSNode;
00353 
00354    // do not insert repeated node
00355    if (n->m_iHeapLoc >= 0)
00356       return;
00357 
00358    m_iLastEntry ++;
00359    m_pHeap[m_iLastEntry] = n;
00360    n->m_llTimeStamp = ts;
00361 
00362    int q = m_iLastEntry;
00363    int p = q;
00364    while (p != 0)
00365    {
00366       p = (q - 1) >> 1;
00367       if (m_pHeap[p]->m_llTimeStamp > m_pHeap[q]->m_llTimeStamp)
00368       {
00369          CSNode* t = m_pHeap[p];
00370          m_pHeap[p] = m_pHeap[q];
00371          m_pHeap[q] = t;
00372          t->m_iHeapLoc = q;
00373          q = p;
00374       }
00375       else
00376          break;
00377    }
00378 
00379    n->m_iHeapLoc = q;
00380 
00381    // first entry, activate the sending queue
00382    if (0 == m_iLastEntry)
00383    {
00384       #ifndef WIN32
00385          pthread_mutex_lock(m_pWindowLock);
00386          pthread_cond_signal(m_pWindowCond);
00387          pthread_mutex_unlock(m_pWindowLock);
00388       #else
00389          SetEvent(*m_pWindowCond);
00390       #endif
00391    }
00392 }

00394 void CSndUList::remove_(const CUDT* u)
00395 {
00396    CSNode* n = u->m_pSNode;
00397 
00398    if (n->m_iHeapLoc >= 0)
00399    {
00400       // remove the node from heap
00401       m_pHeap[n->m_iHeapLoc] = m_pHeap[m_iLastEntry];
00402       m_iLastEntry --;
00403       m_pHeap[n->m_iHeapLoc]->m_iHeapLoc = n->m_iHeapLoc;
00404 
00405       int q = n->m_iHeapLoc;
00406       int p = q * 2 + 1;
00407       while (p <= m_iLastEntry)
00408       {
00409          if ((p + 1 <= m_iLastEntry) && (m_pHeap[p]->m_llTimeStamp > m_pHeap[p + 1]->m_llTimeStamp))
00410             p ++;
00411 
00412          if (m_pHeap[q]->m_llTimeStamp > m_pHeap[p]->m_llTimeStamp)
00413          {
00414             CSNode* t = m_pHeap[p];
00415             m_pHeap[p] = m_pHeap[q];
00416             m_pHeap[p]->m_iHeapLoc = p;
00417             m_pHeap[q] = t;
00418             m_pHeap[q]->m_iHeapLoc = q;
00419 
00420             q = p;
00421             p = q * 2 + 1;
00422          }
00423          else
00424             break;
00425       }
00426 
00427       n->m_iHeapLoc = -1;
00428    }
00429 }

00500 #ifndef WIN32
00501    void* CSndQueue::worker(void* param)
00502 #else
00503    DWORD WINAPI CSndQueue::worker(LPVOID param)
00504 #endif
00505 {
00506    CSndQueue* self = (CSndQueue*)param;
00507 
00508    while (!self->m_bClosing)
00509    {
00510       uint64_t ts = self->m_pSndUList->getNextProcTime();
00511 
00512       if (ts > 0)
00513       {
00514          // wait until next processing time of the first socket on the list
00515          uint64_t currtime;
00516          CTimer::rdtsc(currtime);
00517          if (currtime < ts)
00518             self->m_pTimer->sleepto(ts);
00519 
00520          // it is time to send the next pkt
00521          sockaddr* addr;
00522          CPacket pkt;
00523          if (self->m_pSndUList->pop(addr, pkt) < 0)
00524             continue;
00525 
00526          self->m_pChannel->sendto(addr, pkt);
00527       }
00528       else
00529       {
00530          // wait here if there is no sockets with data to be sent
00531          #ifndef WIN32
00532             pthread_mutex_lock(&self->m_WindowLock);
00533             if (!self->m_bClosing && (self->m_pSndUList->m_iLastEntry < 0))
00534                pthread_cond_wait(&self->m_WindowCond, &self->m_WindowLock);
00535             pthread_mutex_unlock(&self->m_WindowLock);
00536          #else
00537             WaitForSingleObject(self->m_WindowCond, INFINITE);
00538          #endif
00539       }
00540    }
00541 
00542    #ifndef WIN32
00543       return NULL;
00544    #else
00545       SetEvent(self->m_ExitCond);
00546       return 0;
00547    #endif
00548 }

00919 #ifndef WIN32
00920    void* CRcvQueue::worker(void* param)
00921 #else
00922    DWORD WINAPI CRcvQueue::worker(LPVOID param)
00923 #endif
00924 {
00925    CRcvQueue* self = (CRcvQueue*)param;
00926 
00927    sockaddr* addr = (AF_INET == self->m_UnitQueue.m_iIPversion) ? (sockaddr*) new sockaddr_in : (sockaddr*) new sockaddr_in6;
00928    CUDT* u = NULL;
00929    int32_t id;
00930 
00931    while (!self->m_bClosing)
00932    {
00933       #ifdef NO_BUSY_WAITING
00934          self->m_pTimer->tick();
00935       #endif
00936 
00937       // check waiting list, if new socket, insert it to the list
00938       while (self->ifNewEntry())
00939       {
00940          CUDT* ne = self->getNewEntry();
00941          if (NULL != ne)
00942          {
00943             self->m_pRcvUList->insert(ne);
00944             self->m_pHash->insert(ne->m_SocketID, ne);
00945          }
00946       }
00947 
00948       // find next available slot for incoming packet
00949       CUnit* unit = self->m_UnitQueue.getNextAvailUnit();
00950       if (NULL == unit)
00951       {
00952          // no space, skip this packet
00953          CPacket temp;
00954          temp.m_pcData = new char[self->m_iPayloadSize];
00955          temp.setLength(self->m_iPayloadSize);
00956          self->m_pChannel->recvfrom(addr, temp);
00957          delete [] temp.m_pcData;
00958          goto TIMER_CHECK;
00959       }
00960 
00961       unit->m_Packet.setLength(self->m_iPayloadSize);
00962 
00963       // reading next incoming packet, recvfrom returns -1 is nothing has been received
00964       if (self->m_pChannel->recvfrom(addr, unit->m_Packet) < 0)
00965          goto TIMER_CHECK;
00966 
00967       id = unit->m_Packet.m_iID;
00968 
00969       // ID 0 is for connection request, which should be passed to the listening socket or rendezvous sockets
00970       if (0 == id)
00971       {
00972          if (NULL != self->m_pListener)
00973             ((CUDT*)self->m_pListener)->listen(addr, unit->m_Packet);
00974          else if (self->m_pRendezvousQueue->retrieve(addr, id))
00975             self->storePkt(id, unit->m_Packet.clone());
00976       }
00977       else if (id > 0)
00978       {
00979          if (NULL != (u = self->m_pHash->lookup(id)))
00980          {
00981             if (CIPAddress::ipcmp(addr, u->m_pPeerAddr, u->m_iIPversion))
00982             {
00983                if (u->m_bConnected && !u->m_bBroken && !u->m_bClosing)
00984                {
00985                   if (0 == unit->m_Packet.getFlag())
00986                      u->processData(unit);
00987                   else
00988                      u->processCtrl(unit->m_Packet);
00989 
00990                   u->checkTimers();
00991                   self->m_pRcvUList->update(u);
00992                }
00993             }
00994          }
00995          else if (self->m_pRendezvousQueue->retrieve(addr, id))
00996             self->storePkt(id, unit->m_Packet.clone());
00997       }
00998 
00999 TIMER_CHECK:
01000       // take care of the timing event for all UDT sockets
01001 
01002       CRNode* ul = self->m_pRcvUList->m_pUList;
01003       uint64_t currtime;
01004       CTimer::rdtsc(currtime);
01005       uint64_t ctime = currtime - 100000 * CTimer::getCPUFrequency();
01006 
01007       while ((NULL != ul) && (ul->m_llTimeStamp < ctime))
01008       {
01009          CUDT* u = ul->m_pUDT;
01010 
01011          if (u->m_bConnected && !u->m_bBroken && !u->m_bClosing)
01012          {
01013             u->checkTimers();
01014             self->m_pRcvUList->update(u);
01015          }
01016          else
01017          {
01018             // the socket must be removed from Hash table first, then RcvUList
01019             self->m_pHash->remove(u->m_SocketID);
01020             self->m_pRcvUList->remove(u);
01021             u->m_pRNode->m_bOnList = false;
01022          }
01023 
01024          ul = self->m_pRcvUList->m_pUList;
01025       }
01026    }
01027 
01028    if (AF_INET == self->m_UnitQueue.m_iIPversion)
01029       delete (sockaddr_in*)addr;
01030    else
01031       delete (sockaddr_in6*)addr;
01032 
01033    #ifndef WIN32
01034       return NULL;
01035    #else
01036       SetEvent(self->m_ExitCond);
01037       return 0;
01038    #endif

01042 {
01043    CGuard bufferlock(m_PassLock);
01044 
01045    map<int32_t, queue<CPacket*> >::iterator i = m_mBuffer.find(id);
01046 
01047    if (i == m_mBuffer.end())
01048    {
01049       #ifndef WIN32
01050          uint64_t now = CTimer::getTime();
01051          timespec timeout;
01052 
01053          timeout.tv_sec = now / 1000000 + 1;
01054          timeout.tv_nsec = (now % 1000000) * 1000;
01055 
01056          pthread_cond_timedwait(&m_PassCond, &m_PassLock, &timeout);
01057       #else
01058          ReleaseMutex(m_PassLock);
01059          WaitForSingleObject(m_PassCond, 1000);
01060          WaitForSingleObject(m_PassLock, INFINITE);
01061       #endif
01062 
01063       i = m_mBuffer.find(id);
01064       if (i == m_mBuffer.end())
01065       {
01066          packet.setLength(-1);
01067          return -1;
01068       }
01069    }
01070 
01071    // retrieve the earliest packet
01072    CPacket* newpkt = i->second.front();
01073 
01074    if (packet.getLength() < newpkt->getLength())
01075    {
01076       packet.setLength(-1);
01077       return -1;
01078    }
01079 
01080    // copy packet content
01081    memcpy(packet.m_nHeader, newpkt->m_nHeader, CPacket::m_iPktHdrSize);
01082    memcpy(packet.m_pcData, newpkt->m_pcData, newpkt->getLength());
01083    packet.setLength(newpkt->getLength());
01084 
01085    delete [] newpkt->m_pcData;
01086    delete newpkt;
01087 
01088    // remove this message from queue, 
01089    // if no more messages left for this socket, release its data structure
01090    i->second.pop();
01091    if (i->second.empty())
01092       m_mBuffer.erase(i);
01093 
01094    return packet.getLength();
