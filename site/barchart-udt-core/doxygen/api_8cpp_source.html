01133 void CUDTUnited::checkBrokenSockets()
01134 {
01135    CGuard cg(m_ControlLock);
01136 
01137    // set of sockets To Be Closed and To Be Removed
01138    vector<UDTSOCKET> tbc;
01139    vector<UDTSOCKET> tbr;
01140 
01141    for (map<UDTSOCKET, CUDTSocket*>::iterator i = m_Sockets.begin(); i != m_Sockets.end(); ++ i)
01142    {
01143       // check broken connection
01144       if (i->second->m_pUDT->m_bBroken)
01145       {
01146          if (i->second->m_Status == LISTENING)
01147          {
01148             // for a listening socket, it should wait an extra 3 seconds in case a client is connecting
01149             if (CTimer::getTime() - i->second->m_TimeStamp < 3000000)
01150                continue;
01151          }
01152          else if ((i->second->m_pUDT->m_pRcvBuffer->getRcvDataSize() > 0) && (i->second->m_pUDT->m_iBrokenCounter -- > 0))
01153          {
01154             // if there is still data in the receiver buffer, wait longer
01155             continue;
01156          }
01157 
01158          //close broken connections and start removal timer
01159          i->second->m_Status = CLOSED;
01160          i->second->m_TimeStamp = CTimer::getTime();
01161          tbc.push_back(i->first);
01162          m_ClosedSockets[i->first] = i->second;
01163 
01164          // remove from listener's queue
01165          map<UDTSOCKET, CUDTSocket*>::iterator ls = m_Sockets.find(i->second->m_ListenSocket);
01166          if (ls == m_Sockets.end())
01167          {
01168             ls = m_ClosedSockets.find(i->second->m_ListenSocket);
01169             if (ls == m_ClosedSockets.end())
01170                continue;
01171          }
01172 
01173          CGuard::enterCS(ls->second->m_AcceptLock);
01174          ls->second->m_pQueuedSockets->erase(i->second->m_SocketID);
01175          ls->second->m_pAcceptSockets->erase(i->second->m_SocketID);
01176          CGuard::leaveCS(ls->second->m_AcceptLock);
01177       }
01178    }
01179 
01180    for (map<UDTSOCKET, CUDTSocket*>::iterator j = m_ClosedSockets.begin(); j != m_ClosedSockets.end(); ++ j)
01181    {
01182       if (j->second->m_pUDT->m_ullLingerExpiration > 0)
01183       {
01184          // asynchronous close: 
01185          if ((NULL == j->second->m_pUDT->m_pSndBuffer) || (0 == j->second->m_pUDT->m_pSndBuffer->getCurrBufSize()) || (j->second->m_pUDT->m_ullLingerExpiration <= CTimer::getTime()))
01186          {
01187             j->second->m_pUDT->m_ullLingerExpiration = 0;
01188             j->second->m_pUDT->m_bClosing = true;
01189             j->second->m_TimeStamp = CTimer::getTime();
01190          }
01191       }
01192 
01193       // timeout 1 second to destroy a socket AND it has been removed from RcvUList
01194       if ((CTimer::getTime() - j->second->m_TimeStamp > 1000000) && ((NULL == j->second->m_pUDT->m_pRNode) || !j->second->m_pUDT->m_pRNode->m_bOnList))
01195       {
01196          tbr.push_back(j->first);
01197       }
01198    }
01199 
01200    // move closed sockets to the ClosedSockets structure
01201    for (vector<UDTSOCKET>::iterator k = tbc.begin(); k != tbc.end(); ++ k)
01202       m_Sockets.erase(*k);
