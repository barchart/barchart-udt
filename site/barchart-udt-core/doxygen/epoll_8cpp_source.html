00188 int CEPoll::wait(const int eid, set<UDTSOCKET>* readfds, set<UDTSOCKET>* writefds, int64_t msTimeOut, set<SYSSOCKET>* lrfds, set<SYSSOCKET>* lwfds)
00189 {
00190    // if all fields is NULL and waiting time is infinite, then this would be a deadlock
00191    if (!readfds && !writefds && !lrfds && lwfds && (msTimeOut < 0))
00192       throw CUDTException(5, 3, 0);
00193 
00194    int total = 0;
00195 
00196    int64_t entertime = CTimer::getTime();
00197    while (true)
00198    {
00199       CGuard::enterCS(m_EPollLock);
00200 
00201       map<int, CEPollDesc>::iterator p = m_mPolls.find(eid);
00202       if (p == m_mPolls.end())
00203       {
00204          CGuard::leaveCS(m_EPollLock);
00205          throw CUDTException(5, 13);
00206       }
00207 
00208       if (((readfds || writefds) && p->second.m_sUDTSocks.empty()) && ((lrfds || lwfds) && p->second.m_sLocals.empty()))
00209       {
00210          // no socket is being monitored, this may be a deadlock
00211          CGuard::leaveCS(m_EPollLock);
00212          throw CUDTException(5, 3);
00213       }
00214 
00215       if ((NULL != readfds) && !p->second.m_sUDTReads.empty())
00216       {
00217          *readfds = p->second.m_sUDTReads;
00218          total += p->second.m_sUDTReads.size();
00219       }
00220 
00221       if ((NULL != writefds) && !p->second.m_sUDTWrites.empty())
00222       {
00223          *writefds = p->second.m_sUDTWrites;
00224          total += p->second.m_sUDTWrites.size();
00225       }
00226 
00227       if (lrfds || lwfds)
00228       {
00229          if (lrfds)
00230             lrfds->clear();
00231 
00232          if (lwfds)
00233             lwfds->clear();
00234 
00235          #ifdef LINUX
00236          const int max_events = p->second.m_sLocals.size();
00237          epoll_event ev[max_events];
00238          int nfds = epoll_wait(p->second.m_iLocalID, ev, max_events, 0);
00239 
00240          for (int i = 0; i < nfds; ++ i)
00241          {
00242             if ((NULL != lrfds) && (ev[i].events & EPOLLIN))
00243             {
00244                lrfds->insert(ev[i].data.fd);
00245                ++ total;
00246             }
00247             if ((NULL != lwfds) && (ev[i].events & EPOLLOUT))
00248             {
00249                lwfds->insert(ev[i].data.fd);
00250                ++ total;
00251             }
00252          }
00253          #else
00254          //currently "select" is used for all non-Linux platforms.
00255          //faster approaches can be applied for specific systems in the future.
00256 
00257          //"select" has a limitation on the number of sockets
00258 
00259          fd_set readfds;
00260          fd_set writefds;
00261          FD_ZERO(&readfds);
00262          FD_ZERO(&writefds);
00263 
00264          for (set<SYSSOCKET>::const_iterator i = p->second.m_sLocals.begin(); i != p->second.m_sLocals.end(); ++ i)
00265          {
00266             if (lrfds)
00267                FD_SET(*i, &readfds);
00268             if (lwfds)
00269                FD_SET(*i, &writefds);
00270          }
00271 
00272          timeval tv;
00273          tv.tv_sec = 0;
00274          tv.tv_usec = 0;
00275          int r = select(0, &readfds, &writefds, NULL, &tv);
00276 
00277          if (r > 0)
00278          {
00279             for (set<SYSSOCKET>::const_iterator i = p->second.m_sLocals.begin(); i != p->second.m_sLocals.end(); ++ i)
00280             {
00281                if (lrfds)
00282                {
00283                   if (FD_ISSET(*i, &readfds))
00284                   {
00285                      lrfds->insert(*i);
00286                      ++ total;
00287                   }
00288                }
00289 
00290                if (lwfds)
00291                {
00292                   if (FD_ISSET(*i, &writefds))
00293                   {
00294                      lwfds->insert(*i);
00295                      ++ total;
00296                   }
00297                }
00298             }
00299          }
00300          #endif
00301       }
00302 
00303       CGuard::leaveCS(m_EPollLock);
00304 
00305       if (total > 0)
00306          return total;
00307 
00308       if ((msTimeOut >= 0) && (int64_t(CTimer::getTime() - entertime) >= msTimeOut * 1000LL))
00309          break;
00310 
00311       CTimer::waitForEvent();
00312    }
00313 
00314    return 0;
