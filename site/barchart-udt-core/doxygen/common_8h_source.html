00212 // seqcmp: compare two seq#, considering the wraping
00213 // seqlen: length from the 1st to the 2nd seq#, including both
00214 // seqoff: offset from the 2nd to the 1st seq#
00215 // incseq: increase the seq# by 1
00216 // decseq: decrease the seq# by 1
00217 // incseq: increase the seq# by a given offset
00218 
00219 class CSeqNo
00220 {
00221 public:
00222    inline static int seqcmp(const int32_t& seq1, const int32_t& seq2)
00223    {return (abs(seq1 - seq2) < m_iSeqNoTH) ? (seq1 - seq2) : (seq2 - seq1);}
00224 
00225    inline static int seqlen(const int32_t& seq1, const int32_t& seq2)
00226    {return (seq1 <= seq2) ? (seq2 - seq1 + 1) : (seq2 - seq1 + m_iMaxSeqNo + 2);}
00227 
00228    inline static int seqoff(const int32_t& seq1, const int32_t& seq2)
00229    {
00230       if (abs(seq1 - seq2) < m_iSeqNoTH)
00231          return seq2 - seq1;
00232 
00233       if (seq1 < seq2)
00234          return seq2 - seq1 - m_iMaxSeqNo - 1;
00235 
00236       return seq2 - seq1 + m_iMaxSeqNo + 1;
00237    }
00238 
00239    inline static int32_t incseq(const int32_t seq)
00240    {return (seq == m_iMaxSeqNo) ? 0 : seq + 1;}
00241 
00242    inline static int32_t decseq(const int32_t& seq)
00243    {return (seq == 0) ? m_iMaxSeqNo : seq - 1;}
00244 
00245    inline static int32_t incseq(const int32_t& seq, const int32_t& inc)
00246    {return (m_iMaxSeqNo - seq >= inc) ? seq + inc : seq - m_iMaxSeqNo + inc - 1;}
00247 
00248 public:
00249    static const int32_t m_iSeqNoTH;             // threshold for comparing seq. no.
00250    static const int32_t m_iMaxSeqNo;            // maximum sequence number used in UDT
00251 };
