012    /**
013     * UDT channel role type.
014     * 
015     * NOTE: {@link com.barchart.udt#TypeUDT} means stream vs datagram;
016     * {@link KindUDT} means server vs client.
017     */
018    public enum KindUDT {
019    
020            /**
021             * Server mode: listens and accepts connections; generates {@link #COMMUTER}
022             * as a result of {@link SocketUDT#accept()}
023             */
024            ACCEPTOR, //
025    
026            /**
027             * Client mode: user-created channel which initiates connections to servers;
028             * Server mode: channel which is a result of accept();
029             */
030            CONNECTOR, //
031    
032    }
