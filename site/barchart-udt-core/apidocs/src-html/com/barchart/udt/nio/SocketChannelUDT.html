028    /**
029     * {@link SocketChannel}-like wrapper for {@link SocketUDT}, can be either
030     * stream or message oriented, depending on {@link TypeUDT}
031     * <p>
032     * you must use {@link SelectorProviderUDT#openSocketChannel()} to obtain
033     * instance of this class; do not use JDK
034     * {@link java.nio.channels.SocketChannel#open()};
035     * <p>
036     * example:
037     * 
038     * <pre>
039     * SelectorProvider provider = SelectorProviderUDT.DATAGRAM;
040     * SocketChannel clientChannel = provider.openSocketChannel();
041     * clientChannel.configureBlocking(true);
042     * Socket clientSocket = clientChannel.socket();
043     * InetSocketAddress clientAddress = new InetSocketAddress(&quot;localhost&quot;, 10000);
044     * clientSocket.bind(clientAddress);
045     * assert clientSocket.isBound();
046     * InetSocketAddress serverAddress = new InetSocketAddress(&quot;localhost&quot;, 12345);
047     * clientChannel.connect(serverAddress);
048     * assert clientSocket.isConnected();
049     * </pre>
050     */
051    public class SocketChannelUDT extends SocketChannel implements ChannelUDT {
052    
053            protected static final Logger log = LoggerFactory
054                            .getLogger(SocketChannelUDT.class);

082            protected SocketChannelUDT( //
083                            final SelectorProviderUDT provider, //
084                            final SocketUDT socketUDT, //
085                            final boolean isConnected //
086            ) {
087    
088                    super(provider);
089                    this.socketUDT = socketUDT;
090    
091                    if (isConnected) {
092                            isConnectFinished = true;
093                            isConnectionPending = false;
094                    } else {
095                            isConnectFinished = false;
096                            isConnectionPending = true;
097                    }
098    
099            }

101            @Override
102            public boolean connect(final SocketAddress remote) throws IOException {
103    
104                    if (!isOpen()) {
105                            throw new ClosedChannelException();
106                    }
107    
108                    if (isConnected()) {
109                            log.warn("already connected; ignoring remote={}", remote);
110                            return true;
111                    }
112    
113                    if (remote == null) {
114                            close();
115                            log.error("remote == null");
116                            throw new NullPointerException();
117                    }
118    
119                    final InetSocketAddress remoteSocket = (InetSocketAddress) remote;
120    
121                    if (remoteSocket.isUnresolved()) {
122                            log.error("can not use unresolved address: remote={}", remote);
123                            close();
124                            throw new UnresolvedAddressException();
125                    }
126    
127                    if (isBlocking()) {
128                            synchronized (connectLock) {
129                                    try {
130    
131                                            if (isConnectionPending) {
132                                                    close();
133                                                    throw new ConnectionPendingException();
134                                            }
135    
136                                            isConnectionPending = true;
137    
138                                            begin();
139    
140                                            socketUDT.connect(remoteSocket);
141    
142                                    } finally {
143    
144                                            end(true);
145    
146                                            isConnectionPending = false;
147    
148                                            connectLock.notifyAll();
149    
150                                    }
151                            }
152    
153                            return socketUDT.isConnected();
154    
155                    } else {
156    
157                            /** non Blocking */
158    
159                            if (!isRegistered()) {
160    
161                                    /** this channel is independent of any selector */
162    
163                                    log.error("UDT channel is in NON blocking mode; "
164                                                    + "must register with a selector " //
165                                                    + "before trying to connect(); " //
166                                                    + "socketId=" + socketUDT.id());
167    
168                                    throw new IllegalBlockingModeException();
169    
170                            }
171    
172                            /** this channel is registered with a selector */
173    
174                            synchronized (connectLock) {
175    
176                                    if (isConnectionPending) {
177                                            close();
178                                            log.error("connection already in progress");
179                                            throw new ConnectionPendingException();
180                                    }
181    
182                                    isConnectFinished = false;
183                                    isConnectionPending = true;
184    
185                                    socketUDT.connect(remoteSocket);
186    
187                            }
188    
189                            /**
190                             * connection operation must later be completed by invoking the
191                             * #finishConnect() method.
192                             */
193    
194                            return false;
195    
196                    }
197    
198            }
199    
200            @Override
201            public boolean finishConnect() throws IOException {
202    
203                    if (!isOpen()) {
204                            throw new ClosedChannelException();
205                    }
206    
207                    if (isBlocking()) {
208    
209                            synchronized (connectLock) {
210                                    while (isConnectionPending) {
211                                            try {
212                                                    connectLock.wait();
213                                            } catch (final InterruptedException e) {
214                                                    throw new IOException(e);
215                                            }
216                                    }
217                            }
218    
219                    }
220    
221                    if (isConnected()) {
222    
223                            isConnectFinished = true;
224                            isConnectionPending = false;
225    
226                            return true;
227    
228                    } else {
229    
230                            log.error("connect failure : {}", socketUDT);
231                            throw new IOException();
232    
233                    }
234    
235            }

276            /**
277             * See {@link java.nio.channels.SocketChannel#read(ByteBuffer)} contract;
278             * note: this method does not return (-1) as EOS (end of stream flag)
279             * 
280             * @return <code><0</code> should not happen<br>
281             *         <code>=0</code> blocking mode: timeout occurred on receive<br>
282             *         <code>=0</code> non-blocking mode: nothing is received by the
283             *         underlying UDT socket<br>
284             *         <code>>0</code> actual bytes received count<br>
285             * @see com.barchart.udt.SocketUDT#receive(ByteBuffer)
286             * @see com.barchart.udt.SocketUDT#receive(byte[], int, int)
287             */
288            @Override
289            public int read(final ByteBuffer buffer) throws IOException {
290    
291                    final int remaining = buffer.remaining();
292    
293                    if (remaining <= 0) {
294                            return 0;
295                    }
296    
297                    final SocketUDT socket = socketUDT;
298                    final boolean isBlocking = isBlockingMode;
299    
300                    final int sizeReceived;
301    
302                    try {
303    
304                            if (isBlocking) {
305                                    begin(); // JDK contract for NIO blocking calls
306                            }
307    
308                            if (buffer.isDirect()) {
309    
310                                    sizeReceived = socket.receive(buffer);
311    
312                            } else {
313    
314                                    final byte[] array = buffer.array();
315                                    final int position = buffer.position();
316                                    final int limit = buffer.limit();
317    
318                                    sizeReceived = socket.receive(array, position, limit);
319    
320                                    if (0 < sizeReceived && sizeReceived <= remaining) {
321                                            buffer.position(position + sizeReceived);
322                                    }
323    
324                            }
325    
326                    } finally {
327                            if (isBlocking) {
328                                    end(true); // JDK contract for NIO blocking calls
329                            }
330                    }
331    
332                    // see contract for receive()
333    
334                    if (sizeReceived < 0) {
335                            // log.trace("nothing was received; socket={}", socket);
336                            return 0;
337                    }
338    
339                    if (sizeReceived == 0) {
340                            // log.trace("receive timeout; socket={}", socket);
341                            return 0;
342                    }
343    
344                    if (sizeReceived <= remaining) {
345                            return sizeReceived;
346                    } else {
347                            log.error("should not happen: socket={}", socket);
348                            return 0;
349                    }
350    
351            }

381            /**
382             * See {@link java.nio.channels.SocketChannel#write(ByteBuffer)} contract;
383             * 
384             * @return <code><0</code> should not happen<br>
385             *         <code>=0</code> blocking mode: timeout occurred on send<br>
386             *         <code>=0</code> non-blocking mode: buffer is full in the
387             *         underlying UDT socket; nothing is sent<br>
388             *         <code>>0</code> actual bytes sent count<br>
389             * @see com.barchart.udt.SocketUDT#send(ByteBuffer)
390             * @see com.barchart.udt.SocketUDT#send(byte[], int, int)
391             */
392            @Override
393            public int write(final ByteBuffer buffer) throws IOException {
394    
395                    // writeCount.incrementAndGet();
396    
397                    if (buffer == null) {
398                            throw new NullPointerException("buffer == null");
399                    }
400    
401                    final int remaining = buffer.remaining();
402    
403                    if (remaining <= 0) {
404                            return 0;
405                    }
406    
407                    final SocketUDT socket = socketUDT;
408                    final boolean isBlocking = isBlockingMode;
409    
410                    final int sizeSent;
411    
412                    try {
413    
414                            if (isBlocking) {
415                                    begin(); // JDK contract for NIO blocking calls
416                            }
417    
418                            if (buffer.isDirect()) {
419    
420                                    sizeSent = socket.send(buffer);
421    
422                            } else {
423    
424                                    final byte[] array = buffer.array();
425                                    final int position = buffer.position();
426                                    final int limit = buffer.limit();
427    
428                                    sizeSent = socket.send(array, position, limit);
429    
430                                    if (0 < sizeSent && sizeSent <= remaining) {
431                                            buffer.position(position + sizeSent);
432                                    }
433                            }
434                    } finally {
435                            if (isBlocking) {
436                                    end(true); // JDK contract for NIO blocking calls
437                            }
438                    }
439    
440                    // see contract for send()
441    
442                    if (sizeSent < 0) {
443                            // log.trace("no buffer space; socket={}", socket);
444                            return 0;
445                    }
446    
447                    if (sizeSent == 0) {
448                            // log.trace("send timeout; socket={}", socket);
449                            return 0;
450                    }
451    
452                    if (sizeSent <= remaining) {
453                            return sizeSent;
454                    } else {
455                            log.error("should not happen; socket={}", socket);
456                            return 0;
457                    }
458    
459            }

461            @Override
462            public long write(final ByteBuffer[] bufferArray, final int offset,
463                            final int length) throws IOException {
464    
465                    try {
466    
467                            long total = 0;
468    
469                            for (int index = offset; index < offset + length; index++) {
470    
471                                    final ByteBuffer buffer = bufferArray[index];
472    
473                                    final int remaining = buffer.remaining();
474                                    final int processed = write(buffer);
475    
476                                    if (remaining == processed) {
477                                            total += processed;
478                                    } else {
479                                            throw new IllegalStateException(
480                                                            "failed to write buffer in array");
481                                    }
482    
483                            }
484    
485                            return total;
486    
487                    } catch (final Throwable e) {
488                            throw new IOException("failed to write buffer array", e);
489                    }
490    
491            }
