050            protected static Selector open(TypeUDT type) throws IOException {
051                    final SelectorProviderUDT provider;
052                    switch (type) {
053                    case DATAGRAM:
054                            provider = SelectorProviderUDT.DATAGRAM;
055                            break;
056                    case STREAM:
057                            provider = SelectorProviderUDT.STREAM;
058                            break;
059                    default:
060                            log.error("unsupported type={}", type);
061                            throw new IOException("unsupported type");
062                    }
063                    return provider.openSelector();
064            }

068            @Override
069            protected void implCloseSelector() throws IOException {
070                    wakeup();
071                    synchronized (this) {
072                            synchronized (publicRegisteredKeySet) {
073                                    synchronized (publicSelectedKeySet) {
074    
075                                            selectedKeySet.clear();
076    
077                                            for (final SelectionKey key : registeredKeySet) {
078                                                    try {
079                                                            key.channel().close();
080                                                    } catch (Throwable e) {
081                                                            log.error("unexpected", e);
082                                                    }
083                                            }
084    
085                                            registeredKeySet.clear();
086    
087                                    }
088                            }
089                    }
090            }

092            // NOE: register() and select() are blocking each other
093            @Override
094            protected SelectionKey register(AbstractSelectableChannel channel,
095                            int interestOps, Object attachment) {
096    
097                    if (!(channel instanceof ChannelUDT)) {
098                            // also takes care of null
099                            log.error("!(channel instanceof ChannelUDT)");
100                            throw new IllegalSelectorException();
101                    }
102    
103                    synchronized (publicRegisteredKeySet) {
104    
105                            if (publicRegisteredKeySet.size() == maximimSelectorSize) {
106                                    log.error("reached maximimSelectorSize)");
107                                    throw new IllegalSelectorException();
108                            }
109    
110                            ChannelUDT channelUDT = (ChannelUDT) channel;
111    
112                            SocketUDT socketUDT = channelUDT.getSocketUDT();
113    
114                            SelectionKeyUDT keyUDT = new SelectionKeyUDT(//
115                                            this, channelUDT, attachment, interestOps);
116    
117                            // XXX the only place with "add/put"
118                            registeredKeyMap.put(socketUDT.getSocketId(), keyUDT);
119                            registeredKeySet.add(keyUDT);
120    
121                            return keyUDT;
122    
123                    }
124    
125            }

150            @Override
151            // per java.nio.Selector contract 0 input means infinite
152            public int select(long timeout) throws IOException {
153                    if (timeout < 0) {
154                            throw new IllegalArgumentException("Negative timeout");
155                    }
156                    long timeoutUDT = (timeout == 0) ? UDT_TIMEOUT_INFINITE : timeout;
157                    return doSelectInsideLock(timeoutUDT);
158            }

160            @Override
161            public int selectNow() throws IOException {
162                    return doSelectInsideLock(UDT_TIMEOUT_NONE);
163            }

165            /**
166             * Wakeup the selector. NOTE: it is recommended to user timed
167             * {@link #select(long)} (say, timeout = 100 milliseconds) to avoid missed
168             * {@link #wakeup()} calls which are rare but possible in current
169             * implementation (accepted in favor of design simplicity / performance)
170             */
171            @Override
172            public Selector wakeup() {
173                    // publisher for volatile
174                    wakeupStepCount++;
175                    return this;
176            }

178            private volatile int wakeupStepCount;
179    
180            // guarded by doSelectInsideLock
181            private int wakeupBaseCount;
182    
183            private final void saveWakeupBase() {
184                    wakeupBaseCount = wakeupStepCount;
185            }
186    
187            private final boolean isWakeupPending() {
188                    return wakeupBaseCount != wakeupStepCount;
189            }

238            SelectorUDT(SelectorProvider provider, //
239                            int maximumSelectorSize, int maximumConnectorSize) {
240    
241                    super(provider);
242    
243                    registeredKeyMap = new HashMap<Integer, SelectionKeyUDT>();
244    
245                    registeredKeySet = new HashSet<SelectionKeyUDT>();
246                    selectedKeySet = new HashSet<SelectionKeyUDT>();
247                    cancelledKeySet = cancelledKeys();
248    
249                    publicRegisteredKeySet = Collections.unmodifiableSet(registeredKeySet);
250                    publicSelectedKeySet = HelperNIOUDT.ungrowableSet(selectedKeySet);
251    
252                    this.maximimSelectorSize = maximumSelectorSize;
253                    if (isBufferBased) {
254                            readBuffer = SocketUDT.newDirectIntBufer(maximumSelectorSize);
255                            writeBuffer = SocketUDT.newDirectIntBufer(maximumSelectorSize);
256                            exceptBuffer = SocketUDT.newDirectIntBufer(maximumSelectorSize);
257                            sizeBuffer = SocketUDT.newDirectIntBufer(UDT_SIZE_COUNT);
258                            readArray = null;
259                            writeArray = null;
260                            exceptArray = null;
261                            sizeArray = null;
262                    } else {
263                            readBuffer = null;
264                            writeBuffer = null;
265                            exceptBuffer = null;
266                            sizeBuffer = null;
267                            readArray = new int[maximumSelectorSize];
268                            writeArray = new int[maximumSelectorSize];
269                            exceptArray = new int[maximumSelectorSize];
270                            sizeArray = new int[UDT_SIZE_COUNT];
271                    }
272    
273                    this.maximumConnectorSize = maximumConnectorSize;
274                    connectorPool = new ConnectorThreadPoolUDT(maximumConnectorSize);

282            private final int doSelectInsideLock(long millisTimeout) throws IOException {
283                    if (!isOpen()) {
284                            throw new ClosedSelectorException();
285                    }
286                    synchronized (this) {
287                            synchronized (publicRegisteredKeySet) {
288                                    synchronized (publicSelectedKeySet) {
289                                            return doSelectReally(millisTimeout);
290                                    }
291                            }
292                    }
293            }

361                    if (isBufferBased) {
362                            return SocketUDT.select(readBuffer, writeBuffer, exceptBuffer,

366                                            sizeArray, timeout);

407            private final void prepareInterest() {
408    
409                    int readSize = 0;
410                    int writeSize = 0;
411                    int exceptSize = 0;
412    
413                    for (final SelectionKeyUDT keyUDT : registeredKeySet) {
414    
415                            final ChannelUDT channelUDT = keyUDT.channelUDT;
416                            keyUDT.readyOps = 0; // publisher for volatile
417    
418                            if (channelUDT.isOpenSocketUDT()) {
419    
420                                    final int interestOps = keyUDT.interestOps;
421                                    final int socketID = keyUDT.socketID;
422                                    final KindUDT channelType = channelUDT.getChannelKind();
423    
424                                    if ((interestOps & (OP_ACCEPT)) != 0) {
425                                            assert channelType == KindUDT.ACCEPTOR;
426                                            setReadInterest(readSize, socketID);
427                                            readSize++;
428                                    }
429                                    if ((interestOps & (OP_READ)) != 0) {
430                                            assert channelType == KindUDT.CONNECTOR;
431                                            setReadInterest(readSize, socketID);
432                                            readSize++;
433                                    }
434                                    if ((interestOps & (OP_WRITE)) != 0) {
435                                            assert channelType == KindUDT.CONNECTOR;
436                                            setWriteInterest(writeSize, socketID);
437                                            writeSize++;
438                                    }
439                                    if ((interestOps & (OP_CONNECT)) != 0) {
440                                            assert channelType == KindUDT.CONNECTOR;
441                                            assert (interestOps & (OP_READ | OP_WRITE | OP_ACCEPT)) == 0;
442                                            /*
443                                             * UDT does not support select() for connect() operation as
444                                             * yet; using thread pool instead
445                                             */
446                                    }
447                            } else {
448                                    synchronized (cancelledKeySet) {
449                                            cancelledKeySet.add(keyUDT);
450                                    }
451                            }
452                    }
453    
454                    // set sizes
455                    setInterestSize(readSize, writeSize, exceptSize);
456    
457            }
458    
459            private final void processInterest() {
460    
461                    // add ready to selected
462                    updateRead();
463                    updateWrite();
464                    updateExcept();
465    
466            }

468            /**
469             * note: millisTimeout input values contract:
470             * 
471             * -1 : infinite
472             * 
473             * =0 : immediate (but really up to 10 ms UDT resolution slice)
474             * 
475             * >0 : finite
476             */
477            private final int performSelect(long millisTimeout) throws ExceptionUDT {
478                    int readyCount = 0;
479                    saveWakeupBase();
480                    if (millisTimeout < 0) {
481                            /* infinite: do select in slices; check for wakeup; */
482                            do {
483                                    readyCount = doSelectSocketUDT(DEFAULT_MIN_SELECTOR_TIMEOUT);
484                                    if (readyCount > 0 || isWakeupPending()) {
485                                            break;
486                                    }
487                            } while (true);
488                    } else if (millisTimeout > 0) {
489                            /* finite: do select in slices; check for wakeup; count down */
490                            do {
491                                    readyCount = doSelectSocketUDT(DEFAULT_MIN_SELECTOR_TIMEOUT);
492                                    if (readyCount > 0 || isWakeupPending()) {
493                                            break;
494                                    }
495                                    millisTimeout -= DEFAULT_MIN_SELECTOR_TIMEOUT;
496                            } while (millisTimeout > 0);
497                    } else { // millisTimeout == 0
498                            /* immediate: one shot select */
499                            readyCount = doSelectSocketUDT(0);
500                    }
501                    return readyCount;
502            }

504            /**
505             * note: return value contract:
506             * 
507             * <0 : should not happen
508             * 
509             * =0 : means nothing was selected/timeout
510             * 
511             * >0 : number of pending r/w ops, NOT number of selected keys
512             */
513            private final int doSelectReally(long millisTimeout) throws IOException {
514    
515                    selectedKeySet.clear();
516    
517                    processCancelled();
518    
519                    int readyCount = 0;
520    
521                    trySelect: try {
522                            // java.nio.Selector contract for wakeup()
523                            begin();
524                            // pre select
525                            prepareInterest();
526                            // into select
527                            readyCount = performSelect(millisTimeout);
528                            if (readyCount == 0) {
529                                    // timeout, nothing is ready; no need to process post select
530                                    break trySelect;
531                            }
532                            // post select
533                            processInterest();
534                    } finally {
535                            // java.nio.Selector contract for wakeup()
536                            end();
537                    }
538    
539                    // using thread pool based connect() processor
540                    readyCount += updateConnect();
541    
542                    assert readyCount >= 0;
543    
544                    return readyCount;
545    
546            }
547    
548            private final int updateConnect() {
549                    final Queue<SelectionKeyUDT> readyQueue = connectorPool.readyQueue;
550                    if (readyQueue.isEmpty()) {
551                            return 0;
552                    }
553                    int updateCount = 0;
554                    SelectionKeyUDT keyUDT;
555                    while ((keyUDT = readyQueue.poll()) != null) {
556                            // contract:
557                            assert keyUDT.channelUDT.getChannelKind() == KindUDT.CONNECTOR;
558                            assert registeredKeySet.contains(keyUDT);
559                            assert (keyUDT.interestOps & OP_CONNECT) != 0;
560                            assert (keyUDT.interestOps & (OP_READ | OP_WRITE | OP_ACCEPT)) == 0;
561                            //
562                            keyUDT.readyOps |= OP_CONNECT;
563                            selectedKeySet.add(keyUDT);
564                            updateCount++;
565                    }
566                    return updateCount;
567            }

569            private final void updateRead() {
570                    final int readSize = getReadInterestSize();
571                    for (int index = 0; index < readSize; index++) {
572                            final int socketId = getReadInterest(index);
573                            final SelectionKeyUDT keyUDT = registeredKeyMap.get(socketId);
574                            assert keyUDT != null;
575                            switch (keyUDT.channelUDT.getChannelKind()) {
576                            case ACCEPTOR:
577                                    keyUDT.readyOps |= OP_ACCEPT;
578                                    break;
579                            case CONNECTOR:
580                                    keyUDT.readyOps |= OP_READ;
581                                    break;
582                            default:
583                                    assert false : "unexpected default";
584                                    continue;
585                            }
586                            selectedKeySet.add(keyUDT);
587                    }
588            }
589    
590            private final void updateWrite() {
591                    final int writeSize = getWriteInterestSize();
592                    for (int index = 0; index < writeSize; index++) {
593                            final int socketId = getWriteInterest(index);
594                            final SelectionKeyUDT keyUDT = registeredKeyMap.get(socketId);
595                            assert keyUDT != null;
596                            switch (keyUDT.channelUDT.getChannelKind()) {
597                            case ACCEPTOR:
598                                    assert false : "unexpected ACCEPTOR";
599                                    continue;
600                            case CONNECTOR:
601                                    keyUDT.readyOps |= OP_WRITE;
602                                    break;
603                            default:
604                                    assert false : "unexpected default";
605                                    continue;
606                            }
607                            selectedKeySet.add(keyUDT);
608                    }
609            }
610    
611            private final void updateExcept() {
612                    final int exceptSize = getExceptInterestSize();
613                    for (int k = 0; k < exceptSize; k++) {
614                            final int socketId = getExceptInterest(k);
615                            final SelectionKeyUDT keyUDT = registeredKeyMap.get(socketId);
616                            assert keyUDT != null;
617                            switch (keyUDT.channelUDT.getChannelKind()) {
618                            case ACCEPTOR:
619                            case CONNECTOR:
620                                    // set all ready OPS to throw on any operation
621                                    keyUDT.readyOps |= keyUDT.channel().validOps();
622                                    break;
623                            default:
624                                    assert false : "unexpected default";
625                                    continue;
626                            }
627                            selectedKeySet.add(keyUDT);
628                    }
629            }

631            private final void processCancelled() throws IOException {
632                    /*
633                     * Precondition: Synchronized on this, publicRegisteredKeySet,
634                     * publicSelectedKeySet
635                     */
636                    synchronized (cancelledKeySet) {
637                            if (!cancelledKeySet.isEmpty()) {
638                                    for (final SelectionKey key : cancelledKeySet) {
639    
640                                            final SelectionKeyUDT keyUDT = (SelectionKeyUDT) key;
641    
642                                            // XXX the only place with "remove"
643    
644                                            final SelectionKeyUDT removed = //
645                                            registeredKeyMap.remove(keyUDT.socketID);
646                                            assert removed != null;
647    
648                                            final boolean isRemoved = //
649                                            registeredKeySet.remove(keyUDT);
650                                            assert isRemoved;
651    
652                                    }
653                                    cancelledKeySet.clear();
654                            }
655                    }
656            }

658            final void submitConnectRequest(SelectionKeyUDT keyUDT,
659                            InetSocketAddress remote) throws IOException {
660    
661                    // TODO think again if lack of sync is OK?
662    
663                    if (!registeredKeySet.contains(keyUDT)) {
664                            throw new IOException(//
665                                            "connect request while not registered");
666                    }
667    
668                    if ((keyUDT.interestOps & OP_CONNECT) == 0) {
669                            throw new IOException("connect request while not interested;"
670                                            + " key=" + keyUDT);
671                    }
672    
673                    if ((keyUDT.interestOps & (OP_ACCEPT | OP_READ | OP_WRITE)) != 0) {
674                            throw new IOException("connect request while is not sole interest;"
675                                            + " key=" + keyUDT);
676                    }
677    
678                    connectorPool.submitRequest(keyUDT, remote);
679    
680            }
