036    /**
037     * selector
038     * <p>
039     * design guidelines:
040     * <p>
041     * 1) follow general contracts of jdk 6 nio; see <a href=
042     * "https://github.com/barchart/barchart-udt/tree/master/barchart-udt-reference-jdk6"

046     * "https://github.com/netty/netty/blob/master/transport/src/main/java/io/netty/channel/socket/nio/NioEventLoop.java"
047     * >NioEventLoop</a>
048     * <p>
049     * note: you must use {@link SelectorProviderUDT#openSelector()} to obtain

058            /**
059             * use this call to instantiate a selector for UDT
060             */
061            protected static Selector open(final TypeUDT type) throws IOException {
062                    final SelectorProviderUDT provider;
063                    switch (type) {
064                    case DATAGRAM:
065                            provider = SelectorProviderUDT.DATAGRAM;
066                            break;
067                    case STREAM:
068                            provider = SelectorProviderUDT.STREAM;
069                            break;
070                    default:
071                            log.error("unsupported type={}", type);
072                            throw new IOException("unsupported type");
073                    }
074                    return provider.openSelector();
075            }

096             * public view : immutable

098            private final Set<? extends SelectionKey> //

106    
107            /** public view : removal allowed, but not addition */
108            private final Set<? extends SelectionKey> //

167    
168                    for (final SelectionKeyUDT keyUDT : terminatedKeyMap.values()) {
169

176                    }

179                    //

184            /**
185             * @param millisTimeout
186             *            <0 : invinite; =0 : immediate; >0 : finite;
187             */
188            protected int doEpollEnter(final long millisTimeout) throws IOException {
189    
190                    if (!isOpen()) {
191                            log.error("slector is closed");
192                            throw new ClosedSelectorException();
193                    }
194    
195                    try {
196                            selectLock.lock();
197                            return doEpollExclusive(millisTimeout);
198                    } finally {
199                            selectLock.unlock();
200                    }
201    
202            }
203    
204            /**
205             * @param millisTimeout
206             * 
207             *            <0 : invinite;
208             * 
209             *            =0 : immediate;
210             * 
211             *            >0 : finite;
212             * @return
213             * 
214             *         <0 : should not happen
215             * 
216             *         =0 : means nothing was selected/timeout
217             * 
218             *         >0 : number of selected keys
219             */
220    
221            protected int doEpollExclusive(final long millisTimeout) throws IOException {
222    
223                    try {
224    
225                            /** java.nio.Selector contract for wakeup() */
226                            // begin();
227    
228                            /** pre select */
229                            doCancel();
230    
231                            /** select proper */
232                            doEpollSelect(millisTimeout);
233    
234                            /** post select */
235                            doResults();
236    
237                    } finally {
238                            /** java.nio.Selector contract for wakeup() */
239                            // end();
240                    }
241    
242                    return selectedKeyMap.size();
243    
244            }
245    
246            /**
247             * @param millisTimeout
248             * 
249             *            -1 : infinite
250             * 
251             *            =0 : immediate
252             * 
253             *            >0 : finite
254             */
255            protected int doEpollSelect(long millisTimeout) throws ExceptionUDT {
256    
257                    wakeupMarkBase();
258    
259                    int readyCount = 0;
260    
261                    if (millisTimeout < 0) {
262    
263                            /** infinite: do select in slices; check for wakeup; */
264    
265                            do {
266                                    readyCount = doEpollSelectUDT(DEFAULT_MIN_SELECTOR_TIMEOUT);
267                                    if (readyCount > 0 || wakeupIsPending()) {
268                                            break;
269                                    }
270                            } while (true);
271    
272                    } else if (millisTimeout > 0) {
273    
274                            /** finite: do select in slices; check for wakeup; count down */
275    
276                            do {
277                                    readyCount = doEpollSelectUDT(DEFAULT_MIN_SELECTOR_TIMEOUT);
278                                    if (readyCount > 0 || wakeupIsPending()) {
279                                            break;
280                                    }
281                                    millisTimeout -= DEFAULT_MIN_SELECTOR_TIMEOUT;
282                            } while (millisTimeout > 0);
283    
284                    } else {
285    
286                            /** immediate */
287    
288                            readyCount = doEpollSelectUDT(0);
289    
290                    }
291    
292                    return readyCount;
293    
294            }

296            protected int doEpollSelectUDT(final long timeout) throws ExceptionUDT {
297                    return SocketUDT.selectEpoll(//
298                                    epollUDT.id(), //
299                                    readBuffer, //
300                                    writeBuffer, //
301                                    sizeBuffer, //
302                                    timeout //
303                                    );
304            }
305    
306            protected void doResults() {
307    
308                    resultIndex++;
309    
310                    doResultsRead(resultIndex);
311    
312                    doResultsWrite(resultIndex);
313    
314            }
315    
316            protected void doResultsRead(final int resultIndex) {
317    
318                    final int readSize = sizeBuffer.get(UDT_READ_INDEX);
319    
320                    for (int index = 0; index < readSize; index++) {
321    
322                            final int socketId = readBuffer.get(index);
323    
324                            final SelectionKeyUDT keyUDT = registeredKeyMap.get(socketId);
325    
326                            if (keyUDT.doRead(resultIndex)) {
327                                    selectedKeyMap.putIfAbsent(keyUDT, keyUDT);
328                            }
329    
330                    }
331    
332            }
333    
334            protected void doResultsWrite(final int resultIndex) {
335    
336                    final int writeSize = sizeBuffer.get(UDT_WRITE_INDEX);
337    
338                    for (int index = 0; index < writeSize; index++) {
339    
340                            final int socketId = writeBuffer.get(index);
341    
342                            final SelectionKeyUDT keyUDT = registeredKeyMap.get(socketId);
343    
344                            if (keyUDT.doWrite(resultIndex)) {
345                                    selectedKeyMap.putIfAbsent(keyUDT, keyUDT);
346                            }
347    
348                    }
349    
350            }

352            protected EpollUDT epollUDT() {
353                    return epollUDT;
354            }

356            @Override
357            protected void implCloseSelector() throws IOException {
358    
359                    wakeup();
360    
361                    try {
362                            selectLock.lock();
363    
364                            // cancelledKeys().addAll(registeredKeySet);
365    
366                            for (final SelectionKeyUDT keyUDT : registeredKeyMap.values()) {
367                                    terminatedKeyMap.putIfAbsent(keyUDT, keyUDT);
368                            }
369    
370                    } finally {
371                            selectLock.unlock();
372                    }
373    
374                    doCancel();
375    
376            }

378            @SuppressWarnings("unchecked")
379            @Override
380            public Set<SelectionKey> keys() {
381                    if (!isOpen()) {
382                            throw new ClosedSelectorException();
383                    }
384                    return (Set<SelectionKey>) registeredKeySet;
385            }

387            /** 
388             */
389            @Override
390            protected SelectionKey register( //
391                            final AbstractSelectableChannel channel, //
392                            final int interestOps, //
393                            final Object attachment //
394            ) {
395    
396                    if (registeredKeyMap.size() >= maximimSelectorSize) {
397                            log.error("reached maximimSelectorSize");
398                            throw new IllegalSelectorException();
399                    }
400    
401                    if (!(channel instanceof ChannelUDT)) {
402                            log.error("!(channel instanceof ChannelUDT)");
403                            throw new IllegalSelectorException();
404                    }
405    
406                    final ChannelUDT channelUDT = (ChannelUDT) channel;
407    
408                    final Integer socketId = channelUDT.socketUDT().id();
409    
410                    SelectionKeyUDT keyUDT = registeredKeyMap.get(socketId);
411    
412                    if (keyUDT == null) {
413                            keyUDT = new SelectionKeyUDT(this, channelUDT, attachment);
414                            registeredKeyMap.putIfAbsent(socketId, keyUDT);
415                            keyUDT = registeredKeyMap.get(socketId);
416                    }
417    
418                    keyUDT.interestOps(interestOps);
419    
420                    return keyUDT;
421    
422            }
423    
424            @Override
425            public int select() throws IOException {
426                    return select(0);
427            }
428    
429            @Override
430            public int select(final long timeout) throws IOException {
431                    if (timeout < 0) {
432                            throw new IllegalArgumentException("negative timeout");
433                    } else if (timeout > 0) {
434                            return doEpollEnter(timeout);
435                    } else {
436                            return doEpollEnter(SocketUDT.TIMEOUT_INFINITE);
437                    }
438            }

440            @SuppressWarnings("unchecked")
441            @Override
442            public Set<SelectionKey> selectedKeys() {
443                    if (!isOpen()) {
444                            throw new ClosedSelectorException();
445                    }
446                    return (Set<SelectionKey>) selectedKeySet;
447            }
448    
449            @Override
450            public int selectNow() throws IOException {
451                    return doEpollEnter(SocketUDT.TIMEOUT_NONE);
452            }
453    
454            @Override
455            public Selector wakeup() {
456                    wakeupStepCount++;
457                    return this;
458            }
459    
460            protected boolean wakeupIsPending() {
461                    return wakeupBaseCount != wakeupStepCount;
462            }
463    
464            protected void wakeupMarkBase() {
465                    wakeupBaseCount = wakeupStepCount;
466            }
