008    package com.barchart.udt;
009    
010    public enum OptionFormatUDT {
011    
012            DECIMAL() {
013                    @Override
014                    public String convert(Object value) {
015                            if (value instanceof Number) {
016                                    long number = ((Number) value).longValue();
017                                    return String.format("%,d", number);
018                            }
019                            return "invalid value";
020                    }
021            }, //
022    
023            BINARY() {
024                    @Override
025                    public String convert(Object value) {
026                            if (value instanceof Number) {
027                                    long number = ((Number) value).longValue();
028                                    return String.format("%,d (%,d K)", number, number / 1024);
029                            }
030                            return "invalid value";
031                    }
032            }, //
033    
034            BOOLEAN() {
035                    @Override
036                    public String convert(Object value) {
037                            if (value instanceof Boolean) {
038                                    boolean bool = ((Boolean) value).booleanValue();
039                                    return String.format("%b", bool);
040                            }
041                            return "invalid value";
042                    }
043            }, //
044    
045            ;
046    
047            public abstract String convert(Object value);
048    
049            OptionFormatUDT() {
050            }
051    
052    }
