008    package com.barchart.udt.net;
009    
010    import java.io.IOException;
011    import java.net.InetAddress;
012    import java.net.Socket;
013    import java.net.SocketAddress;
014    import java.net.SocketException;
015    import java.net.SocketTimeoutException;
016    import java.nio.channels.ServerSocketChannel;
017    
018    public interface IceServerSocket {
019    
020            /**
021             * 
022             * Binds the <code>ServerSocket</code> to a specific address (IP address and
023             * port number).
024             * <p>
025             * If the address is <code>null</code>, then the system will pick up an
026             * ephemeral port and a valid local address to bind the socket.
027             * <p>
028             * 
029             * @param endpoint
030             *            The IP address & port number to bind to.
031             * @throws IOException
032             *             if the bind operation fails, or if the socket is already
033             *             bound.
034             * @throws SecurityException
035             *             if a <code>SecurityManager</code> is present and its
036             *             <code>checkListen</code> method doesn't allow the operation.
037             * @throws IllegalArgumentException
038             *             if endpoint is a SocketAddress subclass not supported by this
039             *             socket
040             * @since 1.4
041             */
042            void bind(SocketAddress endpoint) throws IOException;
043    
044            /**
045             * 
046             * Binds the <code>ServerSocket</code> to a specific address (IP address and
047             * port number).
048             * <p>
049             * If the address is <code>null</code>, then the system will pick up an
050             * ephemeral port and a valid local address to bind the socket.
051             * <P>
052             * The <code>backlog</code> argument must be a positive value greater than
053             * 0. If the value passed if equal or less than 0, then the default value
054             * will be assumed.
055             * 
056             * @param endpoint
057             *            The IP address & port number to bind to.
058             * @param backlog
059             *            The listen backlog length.
060             * @throws IOException
061             *             if the bind operation fails, or if the socket is already
062             *             bound.
063             * @throws SecurityException
064             *             if a <code>SecurityManager</code> is present and its
065             *             <code>checkListen</code> method doesn't allow the operation.
066             * @throws IllegalArgumentException
067             *             if endpoint is a SocketAddress subclass not supported by this
068             *             socket
069             * @since 1.4
070             */
071            void bind(SocketAddress endpoint, int backlog) throws IOException;
072    
073            /**
074             * Returns the local address of this server socket.
075             * 
076             * @return the address to which this socket is bound, or <code>null</code>
077             *         if the socket is unbound.
078             */
079            InetAddress getInetAddress();
080    
081            /**
082             * Returns the port on which this socket is listening.
083             * 
084             * @return the port number to which this socket is listening or -1 if the
085             *         socket is not bound yet.
086             */
087            int getLocalPort();
088    
089            /**
090             * Returns the address of the endpoint this socket is bound to, or
091             * <code>null</code> if it is not bound yet.
092             * 
093             * @return a <code>SocketAddress</code> representing the local endpoint of
094             *         this socket, or <code>null</code> if it is not bound yet.
095             * @see #getInetAddress()
096             * @see #getLocalPort()
097             * @see #bind(SocketAddress)
098             * @since 1.4
099             */
100    
101            SocketAddress getLocalSocketAddress();
102    
103            /**
104             * Listens for a connection to be made to this socket and accepts it. The
105             * method blocks until a connection is made.
106             * 
107             * <p>
108             * A new Socket <code>s</code> is created and, if there is a security
109             * manager, the security manager's <code>checkAccept</code> method is called
110             * with <code>s.getInetAddress().getHostAddress()</code> and
111             * <code>s.getPort()</code> as its arguments to ensure the operation is
112             * allowed. This could result in a SecurityException.
113             * 
114             * @exception IOException
115             *                if an I/O error occurs when waiting for a connection.
116             * @exception SecurityException
117             *                if a security manager exists and its
118             *                <code>checkAccept</code> method doesn't allow the
119             *                operation.
120             * @exception SocketTimeoutException
121             *                if a timeout was previously set with setSoTimeout and the
122             *                timeout has been reached.
123             * @exception java.nio.channels.IllegalBlockingModeException
124             *                if this socket has an associated channel, the channel is
125             *                in non-blocking mode, and there is no connection ready to
126             *                be accepted
127             * 
128             * @return the new Socket
129             * @see SecurityManager#checkAccept revised 1.4 spec JSR-51
130             */
131            Socket accept() throws IOException;
132    
133            /**
134             * Closes this socket.
135             * 
136             * Any thread currently blocked in {@link #accept()} will throw a
137             * {@link SocketException}.
138             * 
139             * <p>
140             * If this socket has an associated channel then the channel is closed as
141             * well.
142             * 
143             * @exception IOException
144             *                if an I/O error occurs when closing the socket. revised
145             *                1.4 spec JSR-51
146             */
147            void close() throws IOException;
148    
149            /**
150             * Returns the unique {@link java.nio.channels.ServerSocketChannel} object
151             * associated with this socket, if any.
152             * 
153             * <p>
154             * A server socket will have a channel if, and only if, the channel itself
155             * was created via the {@link java.nio.channels.ServerSocketChannel#open
156             * ServerSocketChannel.open} method.
157             * 
158             * @return the server-socket channel associated with this socket, or
159             *         <tt>null</tt> if this socket was not created for a channel
160             * 
161             * @since 1.4 spec JSR-51
162             */
163            ServerSocketChannel getChannel();
164    
165            /**
166             * Returns the binding state of the ServerSocket.
167             * 
168             * @return true if the ServerSocket succesfuly bound to an address
169             * @since 1.4
170             */
171            boolean isBound();
172    
173            /**
174             * Returns the closed state of the ServerSocket.
175             * 
176             * @return true if the socket has been closed
177             * @since 1.4
178             */
179            boolean isClosed();
180    
181            /**
182             * Enable/disable SO_TIMEOUT with the specified timeout, in milliseconds.
183             * With this option set to a non-zero timeout, a call to accept() for this
184             * ServerSocket will block for only this amount of time. If the timeout
185             * expires, a <B>java.net.SocketTimeoutException</B> is raised, though the
186             * ServerSocket is still valid. The option <B>must</B> be enabled prior to
187             * entering the blocking operation to have effect. The timeout must be > 0.
188             * A timeout of zero is interpreted as an infinite timeout.
189             * 
190             * @param timeout
191             *            the specified timeout, in milliseconds
192             * @exception SocketException
193             *                if there is an error in the underlying protocol, such as a
194             *                TCP error.
195             * @since JDK1.1
196             * @see #getSoTimeout()
197             */
198            void setSoTimeout(int timeout) throws SocketException;
199    
200            /**
201             * Retrieve setting for SO_TIMEOUT. 0 returns implies that the option is
202             * disabled (i.e., timeout of infinity).
203             * 
204             * @return the SO_TIMEOUT value
205             * @exception IOException
206             *                if an I/O error occurs
207             * @since JDK1.1
208             * @see #setSoTimeout(int)
209             */
210            int getSoTimeout() throws IOException;
211    
212            /**
213             * Enable/disable the SO_REUSEADDR socket option.
214             * <p>
215             * When a TCP connection is closed the connection may remain in a timeout
216             * state for a period of time after the connection is closed (typically
217             * known as the <tt>TIME_WAIT</tt> state or <tt>2MSL</tt> wait state). For
218             * applications using a well known socket address or port it may not be
219             * possible to bind a socket to the required <tt>SocketAddress</tt> if there
220             * is a connection in the timeout state involving the socket address or
221             * port.
222             * <p>
223             * Enabling <tt>SO_REUSEADDR</tt> prior to binding the socket using
224             * {@link #bind(SocketAddress)} allows the socket to be bound even though a
225             * previous connection is in a timeout state.
226             * <p>
227             * When a <tt>ServerSocket</tt> is created the initial setting of
228             * <tt>SO_REUSEADDR</tt> is not defined. Applications can use
229             * {@link #getReuseAddress()} to determine the initial setting of
230             * <tt>SO_REUSEADDR</tt>.
231             * <p>
232             * The behaviour when <tt>SO_REUSEADDR</tt> is enabled or disabled after a
233             * socket is bound (See {@link #isBound()}) is not defined.
234             * 
235             * @param on
236             *            whether to enable or disable the socket option
237             * @exception SocketException
238             *                if an error occurs enabling or disabling the
239             *                <tt>SO_RESUEADDR</tt> socket option, or the socket is
240             *                closed.
241             * @since 1.4
242             * @see #getReuseAddress()
243             * @see #bind(SocketAddress)
244             * @see #isBound()
245             * @see #isClosed()
246             */
247            void setReuseAddress(boolean on) throws SocketException;
248    
249            /**
250             * Tests if SO_REUSEADDR is enabled.
251             * 
252             * @return a <code>boolean</code> indicating whether or not SO_REUSEADDR is
253             *         enabled.
254             * @exception SocketException
255             *                if there is an error in the underlying protocol, such as a
256             *                TCP error.
257             * @since 1.4
258             * @see #setReuseAddress(boolean)
259             */
260            boolean getReuseAddress() throws SocketException;
261    
262            /**
263             * Returns the implementation address and implementation port of this socket
264             * as a <code>String</code>.
265             * 
266             * @return a string representation of this socket.
267             */
268            String toString();
269    
270            /**
271             * Sets a default proposed value for the SO_RCVBUF option for sockets
272             * accepted from this <tt>ServerSocket</tt>. The value actually set in the
273             * accepted socket must be determined by calling
274             * {@link Socket#getReceiveBufferSize()} after the socket is returned by
275             * {@link #accept()}.
276             * <p>
277             * The value of SO_RCVBUF is used both to set the size of the internal
278             * socket receive buffer, and to set the size of the TCP receive window that
279             * is advertized to the remote peer.
280             * <p>
281             * It is possible to change the value subsequently, by calling
282             * {@link Socket#setReceiveBufferSize(int)}. However, if the application
283             * wishes to allow a receive window larger than 64K bytes, as defined by
284             * RFC1323 then the proposed value must be set in the ServerSocket
285             * <B>before</B> it is bound to a local address. This implies, that the
286             * ServerSocket must be created with the no-argument constructor, then
287             * setReceiveBufferSize() must be called and lastly the ServerSocket is
288             * bound to an address by calling bind().
289             * <p>
290             * Failure to do this will not cause an error, and the buffer size may be
291             * set to the requested value but the TCP receive window in sockets accepted
292             * from this ServerSocket will be no larger than 64K bytes.
293             * 
294             * @exception SocketException
295             *                if there is an error in the underlying protocol, such as a
296             *                TCP error.
297             * 
298             * @param size
299             *            the size to which to set the receive buffer size. This value
300             *            must be greater than 0.
301             * 
302             * @exception IllegalArgumentException
303             *                if the value is 0 or is negative.
304             * 
305             * @since 1.4
306             * @see #getReceiveBufferSize
307             */
308            void setReceiveBufferSize(int size) throws SocketException;
309    
310            /**
311             * Gets the value of the SO_RCVBUF option for this <tt>ServerSocket</tt>,
312             * that is the proposed buffer size that will be used for Sockets accepted
313             * from this <tt>ServerSocket</tt>.
314             * 
315             * <p>
316             * Note, the value actually set in the accepted socket is determined by
317             * calling {@link Socket#getReceiveBufferSize()}.
318             * 
319             * @return the value of the SO_RCVBUF option for this <tt>Socket</tt>.
320             * @exception SocketException
321             *                if there is an error in the underlying protocol, such as a
322             *                TCP error.
323             * @see #setReceiveBufferSize(int)
324             * @since 1.4
325             */
326            int getReceiveBufferSize() throws SocketException;
327    
328            /**
329             * Sets performance preferences for this ServerSocket.
330             * 
331             * <p>
332             * Sockets use the TCP/IP protocol by default. Some implementations may
333             * offer alternative protocols which have different performance
334             * characteristics than TCP/IP. This method allows the application to
335             * express its own preferences as to how these tradeoffs should be made when
336             * the implementation chooses from the available protocols.
337             * 
338             * <p>
339             * Performance preferences are described by three integers whose values
340             * indicate the relative importance of short connection time, low latency,
341             * and high bandwidth. The absolute values of the integers are irrelevant;
342             * in order to choose a protocol the values are simply compared, with larger
343             * values indicating stronger preferences. If the application prefers short
344             * connection time over both low latency and high bandwidth, for example,
345             * then it could invoke this method with the values <tt>(1, 0, 0)</tt>. If
346             * the application prefers high bandwidth above low latency, and low latency
347             * above short connection time, then it could invoke this method with the
348             * values <tt>(0, 1, 2)</tt>.
349             * 
350             * <p>
351             * Invoking this method after this socket has been bound will have no
352             * effect. This implies that in order to use this capability requires the
353             * socket to be created with the no-argument constructor.
354             * 
355             * @param connectionTime
356             *            An <tt>int</tt> expressing the relative importance of a short
357             *            connection time
358             * 
359             * @param latency
360             *            An <tt>int</tt> expressing the relative importance of low
361             *            latency
362             * 
363             * @param bandwidth
364             *            An <tt>int</tt> expressing the relative importance of high
365             *            bandwidth
366             * 
367             * @since 1.5
368             */
369            void setPerformancePreferences(int connectionTime, int latency,
370                            int bandwidth);
