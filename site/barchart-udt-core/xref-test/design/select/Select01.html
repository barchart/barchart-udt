23  /**
24   * possible way to emulate select() with epoll_wait()
25   */
26  public class Select01 extends SocketUDT {
27  
28  	public Select01(final TypeUDT type) throws ExceptionUDT {
29  		super(type);
30  	}
31  
32  	private final static EpollUDT epoll;
33  
34  	static {
35  		try {
36  			epoll = new EpollUDT();
37  		} catch (final ExceptionUDT e) {
38  			log.error("", e);
39  			throw new IllegalStateException(e);
40  		}
41  	}
42  
43  	private static final ConcurrentMap<Integer, Integer> //
44  	registrationMap = new ConcurrentHashMap<Integer, Integer>();
45  
46  	private static void ensureRegistaration(final Integer socketId)
47  			throws ExceptionUDT {
48  		if (registrationMap.containsKey(socketId)) {
49  			return;
50  		} else {
51  			registrationMap.put(socketId, socketId);
52  			epollAdd0(epoll.id(), socketId.intValue(), Opt.BOTH.code);
53  		}
54  	}
55  
56  	private static IntBuffer readBuffer = HelpUDT
57  			.newDirectIntBufer(DEFAULT_MAX_SELECTOR_SIZE);
58  	private static IntBuffer writeBuffer = HelpUDT
59  			.newDirectIntBufer(DEFAULT_MAX_SELECTOR_SIZE);
60  	private static IntBuffer sizeBuffer = HelpUDT
61  			.newDirectIntBufer(UDT_SIZE_COUNT);
62  
63  	/**
64  	 * result of select operation
65  	 * <p>
66  	 * note: exceptions will be reported in both read and write lists
67  	 */
68  	public static class Result {
69  		public final List<Integer> readList = new ArrayList<Integer>();
70  		public final List<Integer> writeList = new ArrayList<Integer>();
71  	}
72  
73  	/**
74  	 * TODO add epoll un-registration
75  	 * 
76  	 * @param socketList
77  	 *            - list of sockets interested in select
78  	 */
79  	public synchronized static Result select( //
80  			final List<Integer> socketList, //
81  			final long millisTimeout) throws ExceptionUDT {
82  
83  		for (final Integer socketId : socketList) {
84  			ensureRegistaration(socketId);
85  		}
86  
87  		epollWait0(//
88  				epoll.id(), readBuffer, writeBuffer, sizeBuffer, millisTimeout);
89  
90  		final int readSize = sizeBuffer.get(UDT_READ_INDEX);
91  
92  		final Result result = new Result();
93  
94  		for (int index = 0; index < readSize; index++) {
95  			result.readList.add(readBuffer.get(index));
96  		}
97  
98  		final int writeSize = sizeBuffer.get(UDT_WRITE_INDEX);
99  
100 		for (int index = 0; index < writeSize; index++) {
101 			result.writeList.add(writeBuffer.get(index));
102 		}
103 
104 		return result;
105 
106 	}
