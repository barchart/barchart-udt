32  /**
33   * perform one way send/recv in 2 threads
34   */
35  public class BenchXferOne {
36  
37  	static final Logger log = LoggerFactory.getLogger(BenchXferOne.class);
38  
39  	/** benchmark duration */
40  	static final int time = 60 * 1000;
41  
42  	/** transfer chunk size */
43  	static final int size = 64 * 1024;
44  
45  	static final Counter benchTime = Metrics.newCounter( //
46  			BenchXferOne.class, "bench time");
47  
48  	static final Counter benchSize = Metrics.newCounter( //
49  			BenchXferOne.class, "bench size");
50  
51  	static {
52  
53  		benchTime.inc(time);
54  
55  		benchSize.inc(size);
56  
57  	}
58  
59  	static final Meter recvRate = Metrics.newMeter( //
60  			BenchXferOne.class, "recv rate", "bytes", TimeUnit.SECONDS);
61  
62  	static final Meter sendRate = Metrics.newMeter( //
63  			BenchXferOne.class, "send rate", "bytes", TimeUnit.SECONDS);
64  
65  	static final Timer recvTime = Metrics.newTimer(BenchXferOne.class,
66  			"recv time", TimeUnit.MILLISECONDS, TimeUnit.SECONDS);
67  
68  	static final Timer sendTime = Metrics.newTimer(BenchXferOne.class,
69  			"send time", TimeUnit.MILLISECONDS, TimeUnit.SECONDS);
70  
71  	public static void main(final String[] args) throws Exception {
72  
73  		log.info("init");
74  
75  		final SocketUDT accept = new SocketUDT(TypeUDT.DATAGRAM);
76  		accept.setBlocking(true);
77  		accept.bind(localSocketAddress());
78  		accept.listen(1);
79  		socketAwait(accept, StatusUDT.LISTENING);
80  		log.info("accept : {}", accept);
81  
82  		final SocketUDT client = new SocketUDT(TypeUDT.DATAGRAM);
83  		client.setBlocking(true);
84  		client.bind(localSocketAddress());
85  		socketAwait(client, StatusUDT.OPENED);
86  		client.connect(accept.getLocalSocketAddress());
87  		socketAwait(client, StatusUDT.CONNECTED);
88  		log.info("client : {}", client);
89  
90  		final SocketUDT server = accept.accept();
91  		server.setBlocking(true);
92  		socketAwait(server, StatusUDT.CONNECTED);
93  		log.info("server : {}", server);
94  
95  		final AtomicBoolean isOn = new AtomicBoolean(true);
96  
97  		final Runnable clientTask = new Runnable() {
98  
99  			@Override
100 			public void run() {
101 				try {
102 
103 					while (isOn.get()) {
104 						runCore();
105 					}
106 
107 				} catch (final Exception e) {
108 					log.error("", e);
109 				}
110 			}
111 
112 			final ByteBuffer buffer = ByteBuffer.allocateDirect(size);
113 
114 			long sequence;
115 
116 			void runCore() throws Exception {
117 
118 				buffer.rewind();
119 				buffer.putLong(0, sequence++);
120 
121 				final TimerContext timer = sendTime.time();
122 
123 				final int count = client.send(buffer);
124 
125 				timer.stop();
126 
127 				if (count != size) {
128 					throw new Exception("count");
129 				}
130 
131 				sendRate.mark(count);
132 
133 			}
134 
135 		};
136 
137 		final Runnable serverTask = new Runnable() {
138 
139 			@Override
140 			public void run() {
141 				try {
142 					while (isOn.get()) {
143 						runCore();
144 					}
145 				} catch (final Exception e) {
146 					log.error("", e);
147 				}
148 			}
149 
150 			final ByteBuffer buffer = ByteBuffer.allocateDirect(size);
151 
152 			long sequence;
153 
154 			void runCore() throws Exception {
155 
156 				buffer.rewind();
157 
158 				final TimerContext timer = recvTime.time();
159 
160 				final int count = server.receive(buffer);
161 
162 				timer.stop();
163 
164 				if (count != size) {
165 					throw new Exception("count");
166 				}
167 
168 				if (this.sequence++ != buffer.getLong(0)) {
169 					throw new Exception("sequence");
170 				}
171 
172 				recvRate.mark(count);
173 
174 			}
175 
176 		};
177 
178 		final ExecutorService executor = Executors.newFixedThreadPool(2);
179 
180 		executor.submit(clientTask);
181 
182 		executor.submit(serverTask);
183 
184 		ConsoleReporterUDT.enable(3, TimeUnit.SECONDS);
185 
186 		Thread.sleep(time);
187 
188 		isOn.set(false);
189 
190 		Thread.sleep(1 * 1000);
191 
192 		executor.shutdownNow();
193 
194 		Metrics.defaultRegistry().shutdown();
195 
196 		accept.close();
197 		client.close();
198 		server.close();
199 
200 		log.info("done");
201 
202 	}
